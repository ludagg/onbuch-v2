\section{Problèmes type Probatoire/Bac}

\sujetbac[Étude d'une fonction trigonométrique]
On considère la fonction $f$ définie sur $\R$ par $f(x) = \cos x + \sqrt{3}\sin x$.
\begin{enumerate}[label=\arabic*)]
    \item Transformer $f(x)$ sous la forme $R\cos(x - \varphi)$ avec $R > 0$ et $\varphi \in [0, 2\pi[$.
    \item Résoudre dans $\R$ l'équation $f(x) = 0$.
    \item Résoudre dans $[0, 2\pi[$ l'inéquation $f(x) \geq 1$.
    \item Déterminer les valeurs maximales et minimales de $f$ sur $\R$.
\end{enumerate}

\sujetbac[Problème de physique]
Un pendule simple oscille suivant la loi horaire $\theta(t) = \theta_0 \cos\left(\omega t + \varphi\right)$, où $\theta$ est l'angle en radians, $t$ le temps en secondes, $\omega > 0$ une constante et $\varphi$ la phase initiale.
\begin{enumerate}[label=\arabic*)]
    \item Sachant que $\theta(0) = \frac{\pi}{6}$ et $\theta'(0) = \frac{\omega\sqrt{3}}{2}$, déterminer $\theta_0$ et $\varphi$ (on rappelle que $\theta'(t) = -\theta_0\omega\sin(\omega t + \varphi)$).
    \item En déduire l'expression de $\theta(t)$ sous la forme $R\cos(\omega t - \psi)$.
    \item Résoudre l'équation $\theta(t) = 0$ pour $t \in [0, \frac{2\pi}{\omega}]$.
\end{enumerate}

\sujetbac[Équation trigonométrique paramétrée]
Soit $m$ un nombre réel. On considère l'équation $(E_m)$ : $\cos x + m\sin x = 1$.
\begin{enumerate}[label=\arabic*)]
    \item Pour $m = 1$, résoudre $(E_1)$ dans $[0, 2\pi[$.
    \item Pour $m = \sqrt{3}$, transformer $\cos x + \sqrt{3}\sin x$ sous la forme $R\cos(x - \varphi)$, puis résoudre $(E_{\sqrt{3}})$ dans $\R$.
    \item Discuter, suivant les valeurs de $m$, le nombre de solutions de $(E_m)$ dans $[0, 2\pi[$.
\end{enumerate}

\section{Corrigés}

\corrige{de l'exercice 1}
On a $\frac{\pi}{12} = \frac{\pi}{3} - \frac{\pi}{4}$.
\begin{align*}
\cos\frac{\pi}{12} &= \cos\left(\frac{\pi}{3} - \frac{\pi}{4}\right) = \cos\frac{\pi}{3}\cos\frac{\pi}{4} + \sin\frac{\pi}{3}\sin\frac{\pi}{4} \\
&= \frac12 \times \frac{\sqrt2}{2} + \frac{\sqrt3}{2} \times \frac{\sqrt2}{2} = \frac{\sqrt2 + \sqrt6}{4}.
\end{align*}
\begin{align*}
\sin\frac{\pi}{12} &= \sin\left(\frac{\pi}{3} - \frac{\pi}{4}\right) = \sin\frac{\pi}{3}\cos\frac{\pi}{4} - \cos\frac{\pi}{3}\sin\frac{\pi}{4} \\
&= \frac{\sqrt3}{2} \times \frac{\sqrt2}{2} - \frac12 \times \frac{\sqrt2}{2} = \frac{\sqrt6 - \sqrt2}{4}.
\end{align*}

\corrige{de l'exercice 2}
\begin{enumerate}[label=\alph*)]
    \item $A = \cos x \cos\frac{\pi}{3} - \sin x \sin\frac{\pi}{3} + \cos x \cos\frac{\pi}{3} + \sin x \sin\frac{\pi}{3} = 2\cos x \cos\frac{\pi}{3} = 2\cos x \times \frac12 = \cos x$.
    \item $B = \sin\frac{\pi}{4}\cos x + \cos\frac{\pi}{4}\sin x - (\sin\frac{\pi}{4}\cos x - \cos\frac{\pi}{4}\sin x) = 2\cos\frac{\pi}{4}\sin x = 2 \times \frac{\sqrt2}{2} \sin x = \sqrt2 \sin x$.
\end{enumerate}

\corrige{de l'exercice 3}
\begin{enumerate}[label=\alph*)]
    \item $\cos^2 x = 1 - \sin^2 x = 1 - \frac{9}{25} = \frac{16}{25}$. Comme $x \in \left]\frac{\pi}{2}, \pi\right[$, $\cos x < 0$, donc $\cos x = -\frac45$.
    \item $\sin(2x) = 2\sin x \cos x = 2 \times \frac35 \times \left(-\frac45\right) = -\frac{24}{25}$.
    $\cos(2x) = \cos^2 x - \sin^2 x = \frac{16}{25} - \frac{9}{25} = \frac{7}{25}$.
\end{enumerate}

\corrige{de l'exercice 4}
\begin{enumerate}[label=\alph*)]
    \item $\cos^4 x - \sin^4 x = (\cos^2 x - \sin^2 x)(\cos^2 x + \sin^2 x) = \cos(2x) \times 1 = \cos(2x)$.
    \item $\frac{\sin(2x)}{1 + \cos(2x)} = \frac{2\sin x \cos x}{1 + (2\cos^2 x - 1)} = \frac{2\sin x \cos x}{2\cos^2 x} = \frac{\sin x}{\cos x} = \tan x$, pour $\cos x \neq 0$.
\end{enumerate}

\corrige{de l'exercice 5}
\begin{enumerate}[label=\alph*)]
    \item $f(x) = \cos x + \sqrt{3}\sin x$. $R = \sqrt{1^2 + (\sqrt{3})^2} = \sqrt{4} = 2$.
    $\cos\varphi = \frac{1}{2}$, $\sin\varphi = \frac{\sqrt{3}}{2}$, donc $\varphi = \frac{\pi}{3}$. Ainsi $f(x) = 2\cos\left(x - \frac{\pi}{3}\right)$.
    \item $g(x) = \sin x - \cos x = -\cos x + \sin x$. $R = \sqrt{(-1)^2 + 1^2} = \sqrt{2}$.
    $\cos\varphi = \frac{-1}{\sqrt{2}} = -\frac{\sqrt2}{2}$, $\sin\varphi = \frac{1}{\sqrt{2}} = \frac{\sqrt2}{2}$, donc $\varphi = \frac{3\pi}{4}$. Ainsi $g(x) = \sqrt{2}\cos\left(x - \frac{3\pi}{4}\right)$.
\end{enumerate}

\corrige{de l'exercice 6}
On a $\cos x + \sqrt{3}\sin x = 2\cos\left(x - \frac{\pi}{3}\right)$. L'équation devient $2\cos\left(x - \frac{\pi}{3}\right) = \sqrt{2}$, soit $\cos\left(x - \frac{\pi}{3}\right) = \frac{\sqrt2}{2}$.
Ainsi $x - \frac{\pi}{3} \equiv \frac{\pi}{4} \pmod{2\pi}$ ou $x - \frac{\pi}{3} \equiv -\frac{\pi}{4} \pmod{2\pi}$.
Donc $x \equiv \frac{7\pi}{12} \pmod{2\pi}$ ou $x \equiv \frac{\pi}{12} \pmod{2\pi}$.

\corrige{de l'exercice 7}
\begin{enumerate}[label=\alph*)]
    \item $\cos x = \frac12 \iff x \equiv \pm\frac{\pi}{3} \pmod{2\pi}$.
    \item $\sin x = -\frac{\sqrt2}{2} \iff x \equiv -\frac{\pi}{4} \pmod{2\pi}$ ou $x \equiv \frac{5\pi}{4} \pmod{2\pi}$.
    \item $\tan x = \sqrt{3} \iff x \equiv \frac{\pi}{3} \pmod{\pi}$.
\end{enumerate}

\corrige{de l'exercice 8}
\begin{enumerate}[label=\alph*)]
    \item $\cos(2x) = \frac12 \iff 2x \equiv \pm\frac{\pi}{3} \pmod{2\pi} \iff x \equiv \pm\frac{\pi}{6} \pmod{\pi}$.
    Dans $[0, 2\pi[$ : $x \in \left\{\frac{\pi}{6}, \frac{5\pi}{6}, \frac{7\pi}{6}, \frac{11\pi}{6}\right\}$.
    \item $\sin\left(3x - \frac{\pi}{4}\right) = \frac{\sqrt3}{2} \iff 3x - \frac{\pi}{4} \equiv \frac{\pi}{3} \pmod{2\pi}$ ou $3x - \frac{\pi}{4} \equiv \frac{2\pi}{3} \pmod{2\pi}$.
    Soit $3x \equiv \frac{7\pi}{12} \pmod{2\pi}$ ou $3x \equiv \frac{11\pi}{12} \pmod{2\pi}$.
    Donc $x \equiv \frac{7\pi}{36} \pmod{\frac{2\pi}{3}}$ ou $x \equiv \frac{11\pi}{36} \pmod{\frac{2\pi}{3}}$.
    Dans $[0, 2\pi[$ : $x \in \left\{\frac{7\pi}{36}, \frac{31\pi}{36}, \frac{55\pi}{36}, \frac{79\pi}{36}, \frac{11\pi}{36}, \frac{35\pi}{36}, \frac{59\pi}{36}, \frac{83\pi}{36}\right\}$.
\end{enumerate}

\corrige{de l'exercice 9}
On pose $X = \cos x$. L'équation devient $2X^2 - 3X + 1 = 0$. $\Delta = 9 - 8 = 1$, donc $X = \frac{3 \pm 1}{4}$, soit $X = 1$ ou $X = \frac12$.
\begin{itemize}
    \item $\cos x = 1 \iff x \equiv 0 \pmod{2\pi}$.
    \item $\cos x = \frac12 \iff x \equiv \pm\frac{\pi}{3} \pmod{2\pi}$.
\end{itemize}
Donc $S = \left\{ \frac{\pi}{3} + 2k\pi, -\frac{\pi}{3} + 2k\pi, 2k\pi \mid k \in \Z \right\}$.

\corrige{de l'exercice 10}
$\sin x + \sin(2x) = 0 \iff \sin x + 2\sin x \cos x = 0 \iff \sin x (1 + 2\cos x) = 0$.
\begin{itemize}
    \item $\sin x = 0 \iff x \equiv 0 \pmod{\pi}$.
    \item $1 + 2\cos x = 0 \iff \cos x = -\frac12 \iff x \equiv \pm\frac{2\pi}{3} \pmod{2\pi}$.
\end{itemize}
Dans $[0, 2\pi[$ : $x \in \left\{0, \pi, \frac{2\pi}{3}, \frac{4\pi}{3}\right\}$.

\corrige{de l'exercice 11}
$\cos x \geq \frac12$. Sur $[0, 2\pi[$, $\cos x = \frac12$ pour $x = \frac{\pi}{3}$ et $x = \frac{5\pi}{3}$.
$\cos x \geq \frac12$ pour $x \in [0, \frac{\pi}{3}] \cup [\frac{5\pi}{3}, 2\pi[$.

\corrige{de l'exercice 12}
$\sin x < \frac{\sqrt2}{2}$. Sur $[-\pi, \pi]$, $\sin x = \frac{\sqrt2}{2}$ pour $x = \frac{\pi}{4}$ et $x = \frac{3\pi}{4}$.
$\sin x < \frac{\sqrt2}{2}$ pour $x \in [-\pi, \frac{\pi}{4}[ \cup ]\frac{3\pi}{4}, \pi]$.
