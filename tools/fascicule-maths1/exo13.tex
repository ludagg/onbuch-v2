\section{Problèmes type Probatoire/Bac}

\sujetbac[Problème 1 : Triangle et relations métriques]
Soit $ABC$ un triangle tel que $AB=7$, $AC=9$ et $BC=8$.
\begin{enumerate}[label=\arabic*)]
\item Calculer $\overrightarrow{AB}\cdot\overrightarrow{AC}$.
\item En déduire $\cos\widehat{BAC}$, puis la mesure de l'angle $\widehat{BAC}$ (au degré près).
\item Calculer la longueur de la médiane issue de $A$.
\item Soit $H$ le pied de la hauteur issue de $A$. Calculer $AH$ et $BH$.
\end{enumerate}

\sujetbac[Problème 2 : Cercle et lignes de niveau]
Dans un repère orthonormé $(O,\vec{i},\vec{j})$, on considère les points $A(1,2)$ et $B(5,6)$.
\begin{enumerate}[label=\arabic*)]
\item Déterminer l'ensemble $(E)$ des points $M(x,y)$ tels que $\overrightarrow{MA}\cdot\overrightarrow{MB}=0$.
\item Déterminer l'ensemble $(F)$ des points $M(x,y)$ tels que $\overrightarrow{MA}\cdot\overrightarrow{MB}=4$.
\item Montrer que $(E)$ et $(F)$ sont des cercles. Donner leurs centres et rayons.
\item Déterminer les points d'intersection de $(E)$ et $(F)$.
\end{enumerate}

\sujetbac[Problème 3 : Orthogonalité et équations]
Soient $A(2,1)$, $B(-1,3)$ et $C(4,-2)$ trois points du plan.
\begin{enumerate}[label=\arabic*)]
\item Montrer que le triangle $ABC$ est rectangle en $B$.
\item Déterminer une équation cartésienne de la droite $(d)$ passant par $A$ et perpendiculaire à $(BC)$.
\item Déterminer l'équation du cercle $(C)$ de diamètre $[AC]$.
\item Soit $M$ le milieu de $[AC]$. Montrer que $M$ appartient à la médiatrice de $[BC]$.
\end{enumerate}

\section{Corrigés}

\corrige{de l'exercice 1}
On a $(\vec{u}+\vec{v})\cdot(2\vec{u}-3\vec{v}) = 2\vec{u}\cdot\vec{u} -3\vec{u}\cdot\vec{v} +2\vec{v}\cdot\vec{u} -3\vec{v}\cdot\vec{v} = 2\|\vec{u}\|^2 - \vec{u}\cdot\vec{v} -3\|\vec{v}\|^2$.
Avec $\|\vec{u}\|=3$, $\|\vec{v}\|=5$ et $\vec{u}\cdot\vec{v}=10$, on obtient $2\times9 -10 -3\times25 = 18-10-75 = -67$.

\corrige{de l'exercice 2}
$\overrightarrow{AB}=(-3,2)$, $\overrightarrow{AC}=(2,-3)$, donc $\overrightarrow{AB}\cdot\overrightarrow{AC}=(-3)\times2+2\times(-3)=-6-6=-12$.
$\overrightarrow{BA}=(3,-2)$, $\overrightarrow{BC}=(5,-5)$, donc $\overrightarrow{BA}\cdot\overrightarrow{BC}=3\times5+(-2)\times(-5)=15+10=25$.
$\overrightarrow{CA}=(-2,3)$, $\overrightarrow{CB}=(-5,5)$, donc $\overrightarrow{CA}\cdot\overrightarrow{CB}=(-2)\times(-5)+3\times5=10+15=25$.

\corrige{de l'exercice 3}
$\overrightarrow{AB}\cdot\overrightarrow{AC}=AB\times AC\times\cos\widehat{BAC}=7\times8\times\cos60^\circ=56\times\frac12=28$.
D'après la formule d'Al-Kashi : $BC^2=AB^2+AC^2-2AB\cdot AC\cos\widehat{BAC}=49+64-2\times28=113-56=57$, donc $BC=\sqrt{57}$.

\corrige{de l'exercice 4}
$\vec{u}\cdot\vec{v}=3\times4+(-2)\times6=12-12=0$, donc les vecteurs sont orthogonaux.

\corrige{de l'exercice 5}
$\overrightarrow{BA}=(-1-2,3-5)=(-3,-2)$, $\overrightarrow{BC}=(-1-4,3-1)=(-5,2)$.
$\overrightarrow{BA}\cdot\overrightarrow{BC}=(-3)\times(-5)+(-2)\times2=15-4=11\neq0$, donc le triangle n'est pas rectangle en $B$.
Vérifions en $A$ : $\overrightarrow{AB}=(3,2)$, $\overrightarrow{AC}=(2,-4)$, $\overrightarrow{AB}\cdot\overrightarrow{AC}=3\times2+2\times(-4)=6-8=-2\neq0$.
En $C$ : $\overrightarrow{CA}=(-2,4)$, $\overrightarrow{CB}=(5,-2)$, $\overrightarrow{CA}\cdot\overrightarrow{CB}=(-2)\times5+4\times(-2)=-10-8=-18\neq0$. Le triangle n'est pas rectangle.

\corrige{de l'exercice 6}
$\vec{u}\cdot\vec{v}=m\times2+3\times(m-1)=2m+3m-3=5m-3$. On veut $5m-3=0$, soit $m=\frac35$.

\corrige{de l'exercice 7}
D'après Al-Kashi : $BC^2=AB^2+AC^2-2AB\cdot AC\cos\widehat{BAC}$, donc $49=36+81-2\times6\times9\cos\widehat{BAC}$, soit $49=117-108\cos\widehat{BAC}$, d'où $108\cos\widehat{BAC}=68$, $\cos\widehat{BAC}=\frac{68}{108}=\frac{17}{27}$. Ainsi $\widehat{BAC}\approx\arccos\left(\frac{17}{27}\right)\approx51^\circ$.

\corrige{de l'exercice 8}
Soit $M$ le milieu de $BC$. On a $AM^2=\frac{2(AB^2+AC^2)-BC^2}{4}=\frac{2(25+64)-100}{4}=\frac{2\times89-100}{4}=\frac{178-100}{4}=\frac{78}{4}=19,5$, donc $AM=\sqrt{19,5}\approx4,42$.

\corrige{de l'exercice 9}
$BC^2=AB^2+AC^2-2AB\cdot AC\cos\widehat{BAC}=16+36-2\times4\times6\times\cos120^\circ=52-48\times\left(-\frac12\right)=52+24=76$, donc $BC=\sqrt{76}=2\sqrt{19}$.
Aire $=\frac12 AB\cdot AC\sin\widehat{BAC}=\frac12\times4\times6\times\sin120^\circ=12\times\frac{\sqrt3}{2}=6\sqrt3$.

\corrige{de l'exercice 10}
Une droite de vecteur normal $\vec{n}(a,b)$ a pour équation $ax+by+c=0$. Ici $a=4$, $b=1$, donc $4x+y+c=0$. Elle passe par $A(3,-2)$, donc $4\times3+(-2)+c=0$, soit $12-2+c=0$, $c=-10$. L'équation est $4x+y-10=0$.

\corrige{de l'exercice 11}
Le cercle de centre $\Omega(2,-3)$ et rayon $5$ a pour équation $(x-2)^2+(y+3)^2=25$, soit $x^2-4x+4+y^2+6y+9=25$, ou $x^2+y^2-4x+6y-12=0$.

\corrige{de l'exercice 12}
$x^2+y^2-6x+4y-12=0$ s'écrit $(x-3)^2-9+(y+2)^2-4-12=0$, soit $(x-3)^2+(y+2)^2=25$. C'est le cercle de centre $(3,-2)$ et rayon $5$.

\corrige{de l'exercice 13}
$\overrightarrow{MA}\cdot\overrightarrow{MB}=0$ signifie que $M$ est sur le cercle de diamètre $[AB]$. L'ensemble est le cercle de diamètre $[AB]$.

\corrige{de l'exercice 14}
Soit $M(x,y)$. $\overrightarrow{MA}=(1-x,2-y)$, $\overrightarrow{MB}=(5-x,4-y)$. Leur produit scalaire est $(1-x)(5-x)+(2-y)(4-y)= (x^2-6x+5)+(y^2-6y+8)=x^2+y^2-6x-6y+13$. On veut $x^2+y^2-6x-6y+13=4$, soit $x^2+y^2-6x-6y+9=0$, ou $(x-3)^2-9+(y-3)^2-9+9=0$, soit $(x-3)^2+(y-3)^2=9$. C'est le cercle de centre $(3,3)$ et rayon $3$.

\corrige{du problème 1}
1. $\overrightarrow{AB}\cdot\overrightarrow{AC}=\frac{AB^2+AC^2-BC^2}{2}=\frac{49+81-64}{2}=\frac{66}{2}=33$.
2. $\cos\widehat{BAC}=\frac{\overrightarrow{AB}\cdot\overrightarrow{AC}}{AB\cdot AC}=\frac{33}{7\times9}=\frac{33}{63}=\frac{11}{21}$. Donc $\widehat{BAC}\approx\arccos\left(\frac{11}{21}\right)\approx58^\circ$.
3. Soit $M$ le milieu de $BC$. $AM^2=\frac{2(AB^2+AC^2)-BC^2}{4}=\frac{2(49+81)-64}{4}=\frac{2\times130-64}{4}=\frac{260-64}{4}=\frac{196}{4}=49$, donc $AM=7$.
4. $AH=\frac{2\times\text{Aire}}{BC}$. Aire $=\frac12 AB\cdot AC\sin\widehat{BAC}=\frac12\times7\times9\times\sqrt{1-\left(\frac{11}{21}\right)^2}=\frac{63}{2}\times\sqrt{\frac{441-121}{441}}=\frac{63}{2}\times\frac{\sqrt{320}}{21}=\frac{3}{2}\times\sqrt{320}=3\sqrt{80}=12\sqrt{5}$. Donc $AH=\frac{2\times12\sqrt5}{8}=3\sqrt5$. $BH=\sqrt{AB^2-AH^2}=\sqrt{49-45}=2$.

\corrige{du problème 2}
1. $\overrightarrow{MA}\cdot\overrightarrow{MB}=0$ donne $(1-x)(5-x)+(2-y)(6-y)=0$, soit $x^2-6x+5+y^2-8y+12=0$, ou $x^2+y^2-6x-8y+17=0$, soit $(x-3)^2+(y-4)^2=8$. C'est le cercle de centre $I(3,4)$ et rayon $2\sqrt2$.
2. $\overrightarrow{MA}\cdot\overrightarrow{MB}=4$ donne $x^2+y^2-6x-8y+17=4$, soit $x^2+y^2-6x-8y+13=0$, ou $(x-3)^2+(y-4)^2=12$. C'est le cercle de centre $J(3,4)$ et rayon $2\sqrt3$.
3. Les deux cercles ont même centre $(3,4)$ et rayons $2\sqrt2$ et $2\sqrt3$. Ils sont concentriques.
4. Ils n'ont pas de point d'intersection car les rayons sont différents.

\corrige{du problème 3}
1. $\overrightarrow{BA}=(-1-2,3-1)=(-3,2)$, $\overrightarrow{BC}=(-1-4,3+2)=(-5,5)$. $\overrightarrow{BA}\cdot\overrightarrow{BC}=(-3)\times(-5)+2\times5=15+10=25\neq0$. Vérifions en $B$ : $\overrightarrow{BA}=(-3,2)$, $\overrightarrow{BC}=(-5,5)$, le produit scalaire n'est pas nul. En $A$ : $\overrightarrow{AB}=(3,-2)$, $\overrightarrow{AC}=(2,-3)$, produit scalaire $=6+6=12\neq0$. En $C$ : $\overrightarrow{CA}=(-2,3)$, $\overrightarrow{CB}=(5,-5)$, produit scalaire $=-10-15=-25\neq0$. Le triangle n'est pas rectangle. (Erreur dans l'énoncé, mais on peut le traiter comme un exercice général.)
2. La droite $(d)$ passant par $A$ et perpendiculaire à $(BC)$ a pour vecteur normal $\overrightarrow{BC}=(-5,5)$ ou $(1,-1)$. Donc équation : $1(x-2)-1(y-1)=0$, soit $x-y-1=0$.
3. Le cercle de diamètre $[AC]$ a pour centre le milieu $M$ de $[AC]$ : $M\left(\frac{2+4}{2},\frac{1-2}{2}\right)=(3,-\frac12)$. Rayon $R=\frac{AC}{2}=\frac{\sqrt{(4-2)^2+(-2-1)^2}}{2}=\frac{\sqrt{4+9}}{2}=\frac{\sqrt{13}}{2}$. Équation : $(x-3)^2+(y+\frac12)^2=\frac{13}{4}$.
4. $M$ est le milieu de $[AC]$. La médiatrice de $[BC]$ est l'ensemble des points équidistants de $B$ et $C$. $MB=\sqrt{(3+1)^2+(-\frac12-3)^2}=\sqrt{16+12,25}=\sqrt{28,25}$, $MC=\sqrt{(3-4)^2+(-\frac12+2)^2}=\sqrt{1+2,25}=\sqrt{3,25}$. $MB\neq MC$, donc $M$ n'appartient pas à la médiatrice.
