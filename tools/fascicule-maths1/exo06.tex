\section{Exercices d'application}

\rubrique{Étude complète d'une fonction polynôme}

\exo{}\diff{1}
Soit $f$ la fonction définie sur $\R$ par $f(x)=x^3-3x^2+2$.
\begin{enumerate}[label=\alph*)]
\item Déterminer les limites de $f$ en $-\infty$ et $+\infty$.
\item Calculer $f'(x)$ et étudier son signe.
\item Dresser le tableau de variations de $f$.
\item Déterminer les coordonnées des points d'intersection de la courbe $\mathcal{C}_f$ avec l'axe des abscisses (on donnera les valeurs exactes).
\item Tracer la courbe $\mathcal{C}_f$ dans un repère orthonormé $(O,\vec{i},\vec{j})$ (unité : 1 cm).
\end{enumerate}

\section{Corrigés}

\corrige{de l'exercice 1}
\begin{enumerate}[label=\alph*)]
\item $\lim_{x\to -\infty} f(x) = -\infty$ et $\lim_{x\to +\infty} f(x) = +\infty$.
\item $f'(x)=3x^2-6x = 3x(x-2)$. $f'(x)=0$ pour $x=0$ ou $x=2$. $f'(x)>0$ sur $]-\infty;0[\cup]2;+\infty[$ et $f'(x)<0$ sur $]0;2[$.
\item Tableau de variations :
\[
\begin{array}{c|ccccccc}
x & -\infty & & 0 & & 2 & & +\infty \\
\hline
f'(x) & & + & 0 & - & 0 & + & \\
\hline
& & 2 & & & & & +\infty \\
f(x) & \nearrow & & \searrow & & \nearrow & \\
& -\infty & & & & -2 & & \\
\end{array}
\]
\item $f(x)=0 \iff x^3-3x^2+2=0$. On remarque que $x=1$ est une racine car $1-3+2=0$. En factorisant : $(x-1)(x^2-2x-2)=0$. Les solutions sont $x=1$, $x=1+\sqrt{3}$ et $x=1-\sqrt{3}$. Les points d'intersection sont $(1;0)$, $(1+\sqrt{3};0)$ et $(1-\sqrt{3};0)$.
\item 
\begin{illus}
\begin{tikzpicture}[scale=0.8]
\begin{axis}[axis lines=center, xmin=-2, xmax=4, ymin=-4, ymax=6, xtick={-2,-1,...,4}, ytick={-4,-2,...,6}, grid=both, xlabel=$x$, ylabel=$y$]
\addplot[domain=-2:4, samples=100, thick, color=blue] {x^3-3*x^2+2};
\addplot[mark=*, only marks, color=red] coordinates {(1,0) (1+sqrt(3),0) (1-sqrt(3),0) (0,2) (2,-2)};
\end{axis}
\end{tikzpicture}
\fig{Courbe de $f$}
\end{illus}
\end{enumerate}

\corrige{de l'exercice 4}
\begin{enumerate}[label=\alph*)]
\item $k(-x)=\dfrac{(-x)^2+1}{(-x)^2-4} = \dfrac{x^2+1}{x^2-4} = k(x)$. Donc $k$ est paire.
\item $\lim_{x\to -\infty} k(x) = 1$, $\lim_{x\to +\infty} k(x) = 1$, $\lim_{x\to -2^-} k(x) = +\infty$, $\lim_{x\to -2^+} k(x) = -\infty$, $\lim_{x\to 2^-} k(x) = -\infty$, $\lim_{x\to 2^+} k(x) = +\infty$.
\item Asymptote horizontale $y=1$ en $-\infty$ et $+\infty$. Asymptotes verticales $x=-2$ et $x=2$.
\item $k'(x)=\dfrac{2x(x^2-4)-2x(x^2+1)}{(x^2-4)^2} = \dfrac{2x^3-8x-2x^3-2x}{(x^2-4)^2} = \dfrac{-10x}{(x^2-4)^2}$. $k'(x)=0$ pour $x=0$. $k'(x)>0$ sur $]-\infty;-2[\cup]-2;0[$ et $k'(x)<0$ sur $]0;2[\cup]2;+\infty[$.
\item Tableau de variations :
\[
\begin{array}{c|ccccccccc}
x & -\infty & & -2 & & 0 & & 2 & & +\infty \\
\hline
