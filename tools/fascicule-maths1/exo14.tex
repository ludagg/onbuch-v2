\section{Exercices d'application}

\rubrique{Repère, coordonnées, distance et milieu}

\exo{}\diff{1}
Dans un repère orthonormé $(O, I, J)$, on donne les points $A(2; -1)$, $B(-3; 4)$ et $C(5; 2)$.
\begin{enumerate}[label=\alph*)]
    \item Calculer les distances $AB$, $AC$ et $BC$.
    \item Le triangle $ABC$ est-il rectangle ? Justifier.
    \item Déterminer les coordonnées du milieu $M$ de $[BC]$.
\end{enumerate}

\exo{}\diff{1}
Soient $A(1; 3)$ et $B(-2; 5)$ deux points du plan.
\begin{enumerate}[label=\alph*)]
    \item Déterminer les coordonnées du point $C$ tel que $B$ soit le milieu de $[AC]$.
    \item Déterminer les coordonnées du point $D$ tel que $ABCD$ soit un parallélogramme.
\end{enumerate}

\exo{}\diff{2}
On considère les points $E(-1; 2)$, $F(3; 4)$ et $G(5; -2)$.
\begin{enumerate}[label=\alph*)]
    \item Montrer que $EFG$ est un triangle isocèle.
    \item Calculer les coordonnées du centre du cercle circonscrit au triangle $EFG$.
\end{enumerate}

\rubrique{Vecteurs et colinéarité}

\exo{}\diff{1}
Dans un repère, on donne $\vec{u}(3; -2)$ et $\vec{v}(-6; 4)$.
\begin{enumerate}[label=\alph*)]
    \item Les vecteurs $\vec{u}$ et $\vec{v}$ sont-ils colinéaires ? Justifier.
    \item Déterminer le réel $k$ tel que $\vec{v} = k\vec{u}$.
\end{enumerate}

\exo{}\diff{2}
Soient $A(2; 1)$, $B(5; 3)$, $C(-1; 4)$ et $D(2; 6)$.
\begin{enumerate}[label=\alph*)]
    \item Montrer que les droites $(AB)$ et $(CD)$ sont parallèles.
    \item Les points $A$, $B$ et $C$ sont-ils alignés ? Justifier.
\end{enumerate}

\exo{}\diff{2}
On donne $P(3; -1)$, $Q(0; 2)$ et $R(-2; 5)$.
\begin{enumerate}[label=\alph*)]
    \item Déterminer les coordonnées du point $S$ tel que $PQRS$ soit un parallélogramme.
    \item Vérifier que les diagonales de $PQRS$ se coupent en leur milieu.
\end{enumerate}

\exo{}\diff{1}

    \item Déterminer le coefficient directeur de $(d)$.
    \item Le point $E(3; 4)$ appartient-il à $(d)$ ?
\end{enumerate}

\exo{}\diff{2}

\begin{enumerate}[label=\alph*)]
    \item Montrer que $(d_1)$ et $(d_2)$ sont sécantes.
    \item Calculer les coordonnées de leur point d'intersection $I$.
\end{enumerate}

\exo{}\diff{2}
Déterminer l'équation réduite de la droite passant par $A(-1; 4)$ et parallèle à la droite $(d) : 3x - y + 2 = 0$.

\exo{}\diff{2}

    \item Déterminer l'équation cartésienne de la droite passant par $C$ et perpendiculaire à $(AB)$.
\end{enumerate}

\section{Problèmes type Probatoire/Bac}

\sujetbac[Problème 1 : Configuration géométrique]
Dans un repère orthonormé $(O, I, J)$, on considère les points $A(2; 1)$, $B(5; 3)$ et $C(-1; 4)$.
\begin{enumerate}[label=\arabic*)]
    \item 
    \begin{enumerate}[label=\alph*)]
        \item Calculer les distances $AB$, $AC$ et $BC$.
        \item Montrer que le triangle $ABC$ est rectangle en $A$.
    \end{enumerate}
    \item 
    \begin{enumerate}[label=\alph*)]
        \item Déterminer les coordonnées du centre $\Omega$ du cercle circonscrit au triangle $ABC$.
        \item Donner le rayon $R$ de ce cercle.
    \end{enumerate}
    \item 
    \begin{enumerate}[label=\alph*)]
        \item Déterminer une équation cartésienne de la droite $(BC)$.
        \item Calculer la distance du point $A$ à la droite $(BC)$.
    \end{enumerate}
    \item Soit $(d)$ la droite passant par $A$ et parallèle à $(BC)$. Donner une équation de $(d)$.
\end{enumerate}

\sujetbac[Problème 2 : Cercle et tangente]
Le plan est muni d'un repère orthonormé. On considère le cercle $(\mathcal{C})$ d'équation $x^2 + y^2 - 2x + 4y - 20 = 0$.
\begin{enumerate}[label=\arabic*)]
    \item 
    \begin{enumerate}[label=\alph*)]
        \item Déterminer le centre $\Omega$ et le rayon $R$ de $(\mathcal{C})$.
        \item Vérifier que le point $A(4; 2)$ appartient à $(\mathcal{C})$.
    \end{enumerate}
    \item 
    \begin{enumerate}[label=\alph*)]
        \item Déterminer une équation de la tangente $(T)$ à $(\mathcal{C})$ au point $A$.
        \item La droite $(T)$ coupe l'axe des abscisses en $B$ et l'axe des ordonnées en $C$. Calculer les coordonnées de $B$ et $C$.
    \end{enumerate}
    \item Soit $(d)$ la droite d'équation $y = 2x - 1$.
    \begin{enumerate}[label=\alph*)]
        \item Montrer que $(d)$ est sécante à $(\mathcal{C})$.
        \item Déterminer les coordonnées des points d'intersection de $(d)$ et $(\mathcal{C})$.
    \end{enumerate}
\end{enumerate}

\sujetbac[Problème 3 : Parallélogramme et cercle]
Dans un repère orthonormé $(O, I, J)$, on donne $A(-2; 1)$, $B(1; 3)$ et $C(4; -1)$.
\begin{enumerate}[label=\arabic*)]
    \item 
    \begin{enumerate}[label=\alph*)]
        \item Déterminer les coordonnées du point $D$ tel que $ABCD$ soit un parallélogramme.
        \item Montrer que $ABCD$ est un rectangle.
    \end{enumerate}
    \item 
    \begin{enumerate}[label=\alph*)]
        \item Déterminer l'équation du cercle $(\mathcal{C})$ circonscrit au rectangle $ABCD$.
        \item Vérifier que $O$ est à l'intérieur de $(\mathcal{C})$.
    \end{enumerate}
    \item Soit $(d)$ la droite passant par $A$ et de vecteur directeur $\vec{v}(2; 1)$.
    \begin{enumerate}[label=\alph*)]
        \item Donner une équation paramétrique de $(d)$.
        \item Déterminer les points d'intersection de $(d)$ avec $(\mathcal{C})$.
    \end{enumerate}
\end{enumerate}

\section{Corrigés}

\corrige{de l'exercice 1}
\begin{enumerate}[label=\alph*)]
    \item $AB = \sqrt{( -3 - 2)^2 + (4 - (-1))^2} = \sqrt{(-5)^2 + 5^2} = \sqrt{25 + 25} = \sqrt{50} = 5\sqrt{2}$.
    $AC = \sqrt{(5 - 2)^2 + (2 - (-1))^2} = \sqrt{3^2 + 3^2} = \sqrt{9 + 9} = \sqrt{18} = 3\sqrt{2}$.
    $BC = \sqrt{(5 - (-3))^2 + (2 - 4)^2} = \sqrt{8^2 + (-2)^2} = \sqrt{64 + 4} = \sqrt{68} = 2\sqrt{17}$.
    \item $AB^2 + AC^2 = 50 + 18 = 68 = BC^2$, donc $ABC$ est rectangle en $A$ d'après la réciproque du théorème de Pythagore.
    \item $M\left(\frac{-3 + 5}{2}; \frac{4 + 2}{2}\right) = (1; 3)$.
\end{enumerate}

\corrige{de l'exercice 2}
\begin{enumerate}[label=\alph*)]
    \item $B$ milieu de $[AC]$ donne $\begin{cases} \frac{x_A + x_C}{2} = x_B \\ \frac{y_A + y_C}{2} = y_B \end{cases}$, soit $\begin{cases} \frac{1 + x_C}{2} = -2 \\ \frac{3 + y_C}{2} = 5 \end{cases}$. Donc $1 + x_C = -4 \Rightarrow x_C = -5$ et $3 + y_C = 10 \Rightarrow y_C = 7$. Ainsi $C(-5; 7)$.
    \item $ABCD$ parallélogramme $\Leftrightarrow \overrightarrow{AB} = \overrightarrow{DC}$. $\overrightarrow{AB}(-3; 2)$. Soit $D(x; y)$, $\overrightarrow{DC}(1 - x; 3 - y)$. On a $\begin{cases} 1 - x = -3 \\ 3 - y = 2 \end{cases} \Rightarrow x = 4, y = 1$. Donc $D(4; 1)$.
\end{enumerate}

\corrige{de l'exercice 3}
\begin{enumerate}[label=\alph*)]
    \item $EF = \sqrt{(3+1)^2 + (4-2)^2} = \sqrt{16 + 4} = \sqrt{20} = 2\sqrt{5}$.
    $FG = \sqrt{(5-3)^2 + (-2-4)^2} = \sqrt{4 + 36} = \sqrt{40} = 2\sqrt{10}$.
    $EG = \sqrt{(5+1)^2 + (-2-2)^2} = \sqrt{36 + 16} = \sqrt{52} = 2\sqrt{13}$.
    Aucun côté égal, donc $EFG$ n'est pas isocèle. (Vérification : $EF = EG$ ? $20 \neq 52$, $EF = FG$ ? $20 \neq 40$, $FG = EG$ ? $40 \neq 52$.)
    \item Le centre du cercle circonscrit est l'intersection des médiatrices. Médiatrice de $[EF]$ : milieu $M_{EF}(1; 3)$, pente de $EF$ : $\frac{4-2}{3+1} = \frac{1}{2}$, donc pente médiatrice $-2$. Équation : $y - 3 = -2(x - 1) \Rightarrow y = -2x + 5$.
    Médiatrice de $[FG]$ : milieu $M_{FG}(4; 1)$, pente de $FG$ : $\frac{-2-4}{5-3} = -3$, pente médiatrice $\frac{1}{3}$. Équation : $y - 1 = \frac{1}{3}(x - 4) \Rightarrow y = \frac{1}{3}x - \frac{1}{3}$.
    Intersection : $-2x + 5 = \frac{1}{3}x - \frac{1}{3} \Rightarrow -6x + 15 = x - 1 \Rightarrow -7x = -16 \Rightarrow x = \frac{16}{7}$, $y = -2\times\frac{16}{7} + 5 = -\frac{32}{7} + \frac{35}{7} = \frac{3}{7}$. Centre $\Omega\left(\frac{16}{7}; \frac{3}{7}\right)$.
\end{enumerate}

\corrige{de l'exercice 4}
\begin{enumerate}[label=\alph*)]
    \item $\vec{u}(3; -2)$ et $\vec{v}(-6; 4)$. $3 \times 4 - (-2) \times (-6) = 12 - 12 = 0$, donc $\vec{u}$ et $\vec{v}$ sont colinéaires.
    \item $\vec{v} = k\vec{u} \Rightarrow (-6; 4) = k(3; -2) \Rightarrow \begin{cases} -6 = 3k \\ 4 = -2k \end{cases} \Rightarrow k = -2$.
\end{enumerate}

\corrige{de l'exercice 5}
\begin{enumerate}[label=\alph*)]
    \item $\overrightarrow{AB}(3; 2)$, $\overrightarrow{CD}(3; 2)$. $\overrightarrow{AB} = \overrightarrow{CD}$, donc $(AB) \parallel (CD)$.
    \item $\overrightarrow{AB}(3; 2)$, $\overrightarrow{AC}(-3; 3)$. $3 \times 3 - 2 \times (-3) = 9 + 6 = 15 \neq 0$, donc $A$, $B$, $C$ ne sont pas alignés.
\end{enumerate}

\corrige{de l'exercice 6}
\begin{enumerate}[label=\alph*)]
    \item $PQRS$ parallélogramme $\Leftrightarrow \overrightarrow{PQ} = \overrightarrow{SR}$. $\overrightarrow{PQ}(-3; 3)$. Soit $S(x; y)$, $\overrightarrow{SR}(-2 - x; 5 - y)$. $\begin{cases} -2 - x = -3 \\ 5 - y = 3 \end{cases} \Rightarrow x = 1, y = 2$. Donc $S(1; 2)$.
    \item Diagonales $[PR]$ et $[QS]$. Milieu de $[PR]$ : $\left(\frac{3-2}{2}; \frac{-1+5}{2}\right) = \left(\frac{1}{2}; 2\right)$. Milieu de $[QS]$ : $\left(\frac{0+1}{2}; \frac{2+2}{2}\right) = \left(\frac{1}{2}; 2\right)$. Même milieu, donc les diagonales se coupent en leur milieu.
\end{enumerate}

\corrige{de l'exercice 7}
Une équation cartésienne de droite passant par $A(x_A; y_A)$ et de vecteur directeur $\vec{v}(a; b)$ est $b(x - x_A) - a(y - y_A) = 0$. Ici $a=4$, $b=1$, $A(2; -3)$ donne $1(x - 2) - 4(y + 3) = 0 \Rightarrow x - 2 - 4y - 12 = 0 \Rightarrow x - 4y - 14 = 0$.

\corrige{de l'exercice 8}
\begin{enumerate}[label=\alph*)]
    \item Un vecteur directeur de $(d) : 2x - 3y + 6 = 0$ est $\vec{v}(3; 2)$.
    \item Coefficient directeur : $m = -\frac{a}{b} = -\frac{2}{-3} = \frac{2}{3}$.
    \item $E(3; 4)$ : $2\times3 - 3\times4 + 6 = 6 - 12 + 6 = 0$, donc $E \in (d)$.
\end{enumerate}

\corrige{de l'exercice 9}
\begin{enumerate}[label=\alph*)]
    \item $(d_1) : y = 2x - 1$ a pour coefficient directeur $2$. $(d_2) : -x + 2y - 3 = 0 \Rightarrow 2y = x + 3 \Rightarrow y = \frac{1}{2}x + \frac{3}{2}$, coefficient directeur $\frac{1}{2}$. $2 \neq \frac{1}{2}$, donc les droites sont sécantes.
    \item Intersection : $2x - 1 = \frac{1}{2}x + \frac{3}{2} \Rightarrow 4x - 2 = x + 3 \Rightarrow 3x = 5 \Rightarrow x = \frac{5}{3}$. $y = 2\times\frac{5}{3} - 1 = \frac{10}{3} - \frac{3}{3} = \frac{7}{3}$. Donc $I\left(\frac{5}{3}; \frac{7}{3}\right)$.
\end{enumerate}

\corrige{de l'exercice 10}
$(d) : 3x - y + 2 = 0$ a pour coefficient directeur $3$. Une droite parallèle a même coefficient directeur $3$. Passant par $A(-1; 4)$ : $y - 4 = 3(x + 1) \Rightarrow y = 3x + 3 + 4 \Rightarrow y = 3x + 7$.

\corrige{de l'exercice 11}
\begin{enumerate}[label=\alph*)]
    \item $\overrightarrow{AB}(2; 3)$. Équation cartésienne de $(AB)$ : $3(x - 1) - 2(y - 2) = 0 \Rightarrow 3x - 3 - 2y + 4 = 0 \Rightarrow 3x - 2y + 1 = 0$.
    \item Vecteur directeur de $(AB)$ : $\vec{v}(2; 3)$. Une droite perpendiculaire a pour vecteur directeur $\vec{w}(-3; 2)$. Passant par $C(-2; 4)$ : $2(x + 2) - (-3)(y - 4) = 0 \Rightarrow 2x + 4 + 3y - 12 = 0 \Rightarrow 2x + 3y - 8 = 0$.
\end{enumerate}

\corrige{de l'exercice 12}
Équation du cercle : $(x - 2)^2 + (y + 3)^2 = 4^2 \Rightarrow (x - 2)^2 + (y + 3)^2 = 16$.

\corrige{de l'exercice 13}
\begin{enumerate}[label=\alph*)]
    \item $x^2 + y^2 - 4x + 6y - 12 = 0 \Rightarrow (x^2 - 4x) + (y^2 + 6y) = 12 \Rightarrow (x - 2)^2 - 4 + (y + 3)^2 - 9 = 12 \Rightarrow (x - 2)^2 + (y + 3)^2 = 25$. C'est un cercle de centre $\Omega(2; -3)$ et de rayon $R = 5$.
    \item $A(5; 1)$ : $(5-2)^2 + (1+3)^2 = 3^2 + 4^2 = 9 + 16 = 25$, donc $A$ appartient au cercle.
\end{enumerate}

\corrige{de l'exercice 14}
Le centre du cercle est le milieu de $[AB]$ : $\Omega\left(\frac{-1+3}{2}; \frac{2+4}{2}\right) = (1; 3)$. Le rayon est $R = \frac{AB}{2} = \frac{\sqrt{(3+1)^2 + (4-2)^2}}{2} = \frac{\sqrt{16 + 4}}{2} = \frac{\sqrt{20}}{2} = \sqrt{5}$. Équation : $(x - 1)^2 + (y - 3)^2 = 5$.

\corrige{de l'exercice 15}
\begin{enumerate}[label=\alph*)]
    \item Centre $\Omega(1; -2)$, rayon $R = 5$. Distance de $\Omega$ à $(d) : 3x - 4y + 9 = 0$ : $d = \frac{|3\times1 - 4\times(-2) + 9|}{\sqrt{3^2 + (-4)^2}} = \frac{|3 + 8 + 9|}{5} = \frac{20}{5} = 4$.
    \item $d = 4 < R = 5$, donc la droite $(d)$ coupe le cercle en deux points.
\end{enumerate}

\corrige{de l'exercice 16}
\begin{enumerate}[label=\alph*)]
    \item $x^2 + y^2 - 6x + 2y - 15 = 0 \Rightarrow (x - 3)^2 - 9 + (y + 1)^2 - 1 = 15 \Rightarrow (x - 3)^2 + (y + 1)^2 = 25$. Centre $\Omega(3; -1)$, rayon $R = 5$.
    \item Distance de $\Omega$ à $(\Delta) : y = 2x - 1 \Rightarrow 2x - y - 1 = 0$. $d = \frac{|2\times3 - (-1) - 1|}{\sqrt{2^2 + (-1)^2}} = \frac{|6 + 1 - 1|}{\sqrt{5}} = \frac{6}{\sqrt{5}} \approx 2,68$. $d < R$, donc $(\Delta)$ coupe $(\Gamma)$ en deux points.
    \item Intersection : $y = 2x - 1$ dans l'équation du cercle : $(x - 3)^2 + (2x - 1 + 1)^2 = 25 \Rightarrow (x - 3)^2 + (2x)^2 = 25 \Rightarrow x^2 - 6x + 9 + 4x^2 = 25 \Rightarrow 5x^2 - 6x - 16 = 0$. Discriminant $\Delta = 36 + 320 = 356 = 4\times89$, $x = \frac{6 \pm 2\sqrt{89}}{10} = \frac{3 \pm \sqrt{89}}{5}$. $y = 2x - 1 = \frac{6 \pm 2\sqrt{89}}{5} - 1 = \frac{1 \pm 2\sqrt{89}}{5}$. Points : $\left(\frac{3 + \sqrt{89}}{5}; \frac{1 + 2\sqrt{89}}{5}\right)$ et $\left(\frac{3 - \sqrt{89}}{5}; \frac{1 - 2\sqrt{89}}{5}\right)$.
\end{enumerate}

\corrige{de l'exercice 17}
Le centre $\Omega$ est sur la perpendiculaire à $(d)$ passant par $T$. Vecteur directeur de $(d) : \vec{v}(2; 1)$, donc vecteur normal $\vec{n}(1; -2)$. La droite $(\Omega T)$ a pour équation paramétrique : $\begin{cases} x = 1 + t \\ y = 1 - 2t \end{cases}$. Soit $\Omega(1 + t; 1 - 2t)$. Rayon $R = \Omega T = \sqrt{t^2 + (-2t)^2} = |t|\sqrt{5}$. De plus, $\Omega A = R$ avec $A(2; 3)$ : $\sqrt{(1 + t - 2)^2 + (1 - 2t - 3)^2} = |t|\sqrt{5} \Rightarrow \sqrt{(t - 1)^2 + (-2t - 2)^2} = |t|\sqrt{5}$. Élevons au carré : $(t - 1)^2 + (2t + 2)^2 = 5t^2 \Rightarrow t^2 - 2t + 1 + 4t^2 + 8t + 4 = 5t^2 \Rightarrow 5t^2 + 6t + 5 = 5t^2 \Rightarrow 6t + 5 = 0 \Rightarrow t = -\frac{5}{6}$. Alors $\Omega\left(1 - \frac{5}{6}; 1 + \frac{10}{6}\right) = \left(\frac{1}{6}; \frac{16}{6}\right) = \left(\frac{1}{6}; \frac{8}{3}\right)$, $R = \frac{5}{6}\sqrt{5}$. \corrige{du problème 3}
\begin{enumerate}[label=\arabic*)]
    \item 
    \begin{enumerate}[label=\alph*)]
        \item $ABCD$ parallélogramme $\Leftrightarrow \overrightarrow{AB} = \overrightarrow{DC}$. $\overrightarrow{AB}(3; 2)$. Soit $D(x; y)$, $\overrightarrow{DC}(4 - x; -1 - y)$. $\begin{cases} 4 - x = 3 \\ -1 - y = 2 \end{cases} \Rightarrow x = 1, y = -3$. Donc $D(1; -3)$.

Produit scalaire $\overrightarrow{AB} \cdot \overrightarrow{BC} = 3\times3 + 2\times(-4) = 9 - 8 = 1 \neq 0$, donc $ABCD$ n'est pas un rectangle.
