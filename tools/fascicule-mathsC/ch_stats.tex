\chapter{Statistiques}

\noindent\textit{La statistique organise, résume et interprète des données issues
de l'observation : notes d'une classe, prix au marché, superficies cultivées,
populations de villes. On apprend d'abord à \textbf{résumer} une série d'une seule
variable par quelques nombres clés (moyenne, médiane, quartiles, variance,
écart-type). Puis, lorsqu'on observe \emph{deux} caractères à la fois, on cherche
s'ils sont \textbf{liés} : nuage de points, point moyen, covariance, corrélation,
et — si le lien est quasi affine — une \textbf{droite d'ajustement} qui permet de
\textbf{prévoir}.}
\vspace{8pt}

% ═══════════════════════════════════════════════════════════════
\section{Vocabulaire de la statistique}

\begin{definition}[Population, individu, caractère]
La \textbf{population} est l'ensemble des objets étudiés ; chacun est un
\textbf{individu} (ou unité statistique). On observe sur chaque individu un
\textbf{caractère} (ou variable). Le caractère est \textbf{qualitatif} s'il n'est
pas un nombre (sexe, langue, mention), \textbf{quantitatif} sinon ; il est
\textbf{discret} si ses valeurs sont isolées (nombre d'enfants, note sur $20$),
\textbf{continu} s'il prend toute valeur d'un intervalle (taille, masse), auquel
cas on regroupe les données en \textbf{classes}.
\end{definition}

\begin{definition}[Effectifs, fréquences]
Pour une série discrète prenant les valeurs $x_1,x_2,\dots,x_p$ avec les
\textbf{effectifs} $n_1,n_2,\dots,n_p$, l'\textbf{effectif total} est
$N=\sum_{i=1}^{p}n_i$. La \textbf{fréquence} de la valeur $x_i$ est
\[
f_i=\frac{n_i}{N},\qquad\text{avec}\qquad \sum_{i=1}^{p}f_i=1 .
\]
L'\textbf{effectif cumulé croissant} (ECC) en $x_i$ est $n_1+\dots+n_i$ ; la
\textbf{fréquence cumulée croissante} (FCC) en $x_i$ est $f_1+\dots+f_i$.
\end{definition}

\begin{remarque}
Pour un caractère continu regroupé en classes $[a_{i-1},a_i[$, on associe à chaque
classe son \textbf{centre} $c_i=\dfrac{a_{i-1}+a_i}{2}$ et son \textbf{amplitude}
$a_i-a_{i-1}$. Tous les calculs (moyenne, variance\dots) se font alors sur les
centres de classes, ce qui introduit une légère approximation.
\end{remarque}

\section{Représentations graphiques}

\begin{definition}[Diagramme en bâtons, histogramme]
Pour un caractère \textbf{discret}, on représente les effectifs (ou fréquences)
par un \textbf{diagramme en bâtons} : à chaque valeur $x_i$, un bâton de hauteur
$n_i$. Pour un caractère \textbf{continu} regroupé en classes, on trace un
\textbf{histogramme} : sur chaque classe, un rectangle dont l'\emph{aire} est
proportionnelle à l'effectif. Si toutes les classes ont la même amplitude, la
hauteur est simplement proportionnelle à l'effectif.
\end{definition}

\begin{aretenir}
\textbf{Le piège de l'histogramme} : c'est l'\textbf{aire} du rectangle qui
représente l'effectif, pas sa hauteur. Quand les amplitudes sont inégales, on
calcule la \textbf{hauteur} (densité) $h_i=\dfrac{n_i}{a_i-a_{i-1}}$.
\end{aretenir}

\section{Caractéristiques de position}

\begin{definition}[Moyenne arithmétique]
La \textbf{moyenne} d'une série de valeurs $x_i$ d'effectifs $n_i$ ($N=\sum n_i$) est
\[
\bar x=\frac1N\sum_{i=1}^{p}n_i\,x_i=\sum_{i=1}^{p}f_i\,x_i .
\]
Pour une série brute de $N$ valeurs (chaque effectif vaut $1$),
$\bar x=\dfrac1N\sum_{i=1}^{N}x_i$.
\end{definition}

\begin{propriete}[Linéarité de la moyenne]
Si l'on transforme chaque donnée par $y_i=a\,x_i+b$ ($a,b$ réels), alors la nouvelle
moyenne vaut $\bar y=a\,\bar x+b$. (Le changement d'origine et d'échelle se transmet
à la moyenne.)
\end{propriete}

\begin{definition}[Médiane]
Une \textbf{médiane} $M_e$ partage la série \emph{ordonnée} en deux parties de même
effectif : au moins la moitié des valeurs lui sont inférieures ou égales, au moins
la moitié supérieures ou égales. Pour $N$ valeurs ordonnées :
\begin{itemize}
\item si $N$ est \textbf{impair}, $M_e$ est la valeur de rang $\dfrac{N+1}{2}$ ;
\item si $N$ est \textbf{pair}, on prend $M_e=\dfrac{1}{2}\big(v_{N/2}+v_{N/2+1}\big)$
(moyenne des deux valeurs centrales).
\end{itemize}
\end{definition}

\begin{remarque}
La médiane est \textbf{robuste} : contrairement à la moyenne, elle n'est pas affectée
par les valeurs extrêmes. Sur les revenus d'un quartier, une seule fortune fait
exploser la moyenne mais ne déplace presque pas la médiane.
\end{remarque}

\begin{definition}[Mode]
Le \textbf{mode} est la valeur (ou la classe modale) de plus grand effectif. Une série
peut admettre plusieurs modes.
\end{definition}

\begin{definition}[Quartiles]
Les \textbf{quartiles} $Q_1$, $Q_2=M_e$, $Q_3$ partagent la série ordonnée en quatre
parties d'effectifs sensiblement égaux. $Q_1$ (premier quartile) est la plus petite
valeur telle qu'au moins $25\,\%$ des données lui soient inférieures ou égales ; $Q_3$
(troisième quartile) telle qu'au moins $75\,\%$ le soient. L'\textbf{écart
interquartile} est $Q_3-Q_1$.
\end{definition}

\begin{aretenir}
\textbf{Méthode pratique des quartiles (effectif $N$).} On ordonne la série.
Le rang de $Q_1$ est $\dfrac N4$, celui de $Q_3$ est $\dfrac{3N}4$ : si ce rang
n'est pas entier on \textbf{arrondit à l'entier supérieur} et on prend la valeur
de ce rang ; s'il est entier, on prend la moyenne de cette valeur et de la suivante.
$Q_2$ est la médiane.
\end{aretenir}

\section{Caractéristiques de dispersion}

\begin{definition}[Variance, écart-type]
La \textbf{variance} d'une série de moyenne $\bar x$ est la moyenne des carrés des
écarts à la moyenne :
\[
V(X)=\frac1N\sum_{i=1}^{p}n_i\,(x_i-\bar x)^2 .
\]
L'\textbf{écart-type} est $\sigma_X=\sqrt{V(X)}$. Il mesure la dispersion des
données autour de la moyenne et s'exprime dans la \emph{même unité} que $X$.
\end{definition}

\begin{theoreme}[Formule de König--Huygens]
Pour toute série statistique,
\[
\boxed{\,V(X)=\frac1N\sum_{i=1}^{p}n_i\,x_i^{\,2}-\bar x^{\,2}=\overline{x^2}-\bar x^{\,2}\,}
\]
(« moyenne des carrés moins carré de la moyenne »). C'est la forme la plus commode
pour le calcul.
\end{theoreme}

\begin{remarque}[Preuve de König]
En développant le carré :
$\dfrac1N\sum n_i(x_i-\bar x)^2=\dfrac1N\sum n_i\big(x_i^2-2\bar x\,x_i+\bar x^2\big)
=\overline{x^2}-2\bar x\cdot\bar x+\bar x^2=\overline{x^2}-\bar x^2$,
car $\dfrac1N\sum n_i x_i=\bar x$ et $\dfrac1N\sum n_i=1$.
\end{remarque}

\begin{propriete}[Effet d'une transformation affine]
Si $y_i=a\,x_i+b$, alors
\[
\bar y=a\bar x+b,\qquad V(Y)=a^2\,V(X),\qquad \sigma_Y=|a|\,\sigma_X .
\]
La variance \emph{ignore} la translation $b$ et est multipliée par $a^2$.
\end{propriete}

\begin{exemple}
Série $\;4,\ 7,\ 7,\ 10,\ 12$ ($N=5$).
$\bar x=\dfrac{4+7+7+10+12}{5}=\dfrac{40}{5}=8$.
$\overline{x^2}=\dfrac{16+49+49+100+144}{5}=\dfrac{358}{5}=71{,}6$.
$V(X)=71{,}6-8^2=71{,}6-64=7{,}6$, donc $\sigma_X=\sqrt{7{,}6}\approx2{,}76$.
\end{exemple}

\begin{aretenir}
\textbf{Mémo dispersion.} Petit écart-type $\Rightarrow$ données resserrées autour
de $\bar x$ ; grand écart-type $\Rightarrow$ données étalées. Toujours vérifier
$V(X)\ge0$ : un résultat négatif révèle une erreur de calcul (souvent un carré
oublié dans König).
\end{aretenir}

\section{Série double, nuage de points et point moyen}

\begin{definition}[Série statistique double]
Une \textbf{série statistique double} est la donnée, pour chaque individu $i$ ($1\le i\le n$),
d'un couple $(x_i,y_i)$ de valeurs de deux caractères quantitatifs $X$ et $Y$. Représentée
dans un repère, elle forme un \textbf{nuage de points} $M_i(x_i\,;\,y_i)$.
\end{definition}

\begin{definition}[Séries marginales et point moyen]
Les séries $(x_i)$ et $(y_i)$ prises séparément sont les \textbf{séries marginales}. Leurs
moyennes sont
\[
\bar x=\frac1n\sum_{i=1}^{n}x_i,\qquad \bar y=\frac1n\sum_{i=1}^{n}y_i.
\]
Le \textbf{point moyen} du nuage est $G(\bar x\,;\,\bar y)$. C'est le « centre de gravité » des points.
\end{definition}

\begin{remarque}
Pour chaque série marginale on dispose aussi de sa \textbf{variance} et de son
\textbf{écart-type}, calculés par König :
\[
V(X)=\frac1n\sum_{i=1}^{n}x_i^{2}-\bar x^{\,2},\qquad \sigma_X=\sqrt{V(X)} .
\]
\end{remarque}

\section{Covariance et coefficient de corrélation}

\begin{definition}[Covariance]
La \textbf{covariance} de $X$ et $Y$ est
\[
\mathrm{cov}(X,Y)=\frac1n\sum_{i=1}^{n}(x_i-\bar x)(y_i-\bar y)
=\frac1n\sum_{i=1}^{n}x_iy_i-\bar x\,\bar y .
\]
La seconde écriture (« de König ») est la plus commode pour les calculs.
\end{definition}

\begin{remarque}
La covariance a le signe de la « tendance » du nuage : positive s'il monte, négative s'il
descend, proche de $0$ si aucune direction ne se dégage. On a toujours $\mathrm{cov}(X,X)=V(X)$.
\end{remarque}

\begin{definition}[Coefficient de corrélation linéaire]
On appelle \textbf{coefficient de corrélation linéaire} de $X$ et $Y$ le nombre
\[
r=\frac{\mathrm{cov}(X,Y)}{\sigma_X\,\sigma_Y}.
\]
\end{definition}

\begin{propriete}[Encadrement et interprétation]
Pour toute série double, $\boxed{-1\le r\le 1}$. De plus :
\begin{itemize}
\item $r$ a le même signe que $\mathrm{cov}(X,Y)$ ;
\item $|r|=1$ équivaut à un \textbf{alignement parfait} des points ;
\item en pratique, on considère l'ajustement affine \textbf{justifié} lorsque $|r|$ est
proche de $1$ (souvent $|r|\ge 0{,}87$, soit $r^2\ge 0{,}75$).
\end{itemize}
\end{propriete}

\begin{aretenir}
Un coefficient $r$ proche de $\pm1$ signale seulement une \textbf{liaison linéaire} forte :
corrélation \emph{n'est pas} causalité. On ajuste par une droite uniquement quand $|r|$ est
proche de $1$.
\end{aretenir}

\section{Droites de régression (moindres carrés)}

\begin{definition}[Méthode des moindres carrés]
La \textbf{droite de régression de $y$ en $x$} est la droite $y=ax+b$ rendant minimale la
somme des carrés des écarts verticaux $\sum_{i}\big(y_i-(ax_i+b)\big)^2$ entre les points et
la droite.
\end{definition}

\begin{theoreme}[Droite de $y$ en $x$]
La droite de régression de $y$ en $x$ a pour coefficients
\[
a=\frac{\mathrm{cov}(X,Y)}{V(X)},\qquad b=\bar y-a\,\bar x,
\]
et elle \textbf{passe toujours par le point moyen} $G(\bar x\,;\,\bar y)$.
\end{theoreme}

\begin{remarque}[Idée de la démonstration]
On note $S(a,b)=\sum_i(y_i-ax_i-b)^2$. Pour $a$ fixé, $S$ est minimale en $b$ quand
$\dfrac{\partial S}{\partial b}=-2\sum_i(y_i-ax_i-b)=0$, soit $b=\bar y-a\bar x$ : la droite
passe par $G$. En reportant et en minimisant en $a$, on obtient
$a=\dfrac{\mathrm{cov}(X,Y)}{V(X)}$. (Le détail dérivatif n'est pas exigible, mais le
\emph{passage par $G$} l'est.)
\end{remarque}

\begin{propriete}[Droite de $x$ en $y$]
De façon symétrique, la droite de régression de $x$ en $y$ s'écrit $x=a'y+b'$ avec
\[
a'=\frac{\mathrm{cov}(X,Y)}{V(Y)},\qquad b'=\bar x-a'\,\bar y .
\]
Elle passe aussi par $G$. Les deux droites se coupent donc en $G$, et l'on a la relation
$r^{2}=a\,a'$.
\end{propriete}

\begin{remarque}
Les deux droites sont en général distinctes ; elles coïncident exactement lorsque $|r|=1$
(alignement parfait). Pour \textbf{prévoir} $y$ à partir de $x$, on utilise toujours la droite
\emph{de $y$ en $x$}.
\end{remarque}

\section{Ajustement par changement de variable}

\begin{aretenir}
Quand le nuage n'est \emph{pas} linéaire mais semble \textbf{exponentiel} ($y\approx k\,e^{mx}$)
ou \textbf{puissance/logarithmique}, on se ramène à un ajustement affine par un
\textbf{changement de variable} :
\[
y=k\,e^{mx}\;\Longrightarrow\;\underbrace{\ln y}_{Z}=m\,x+\ln k\quad(\text{affine en }x);
\qquad
y=k\,x^{\alpha}\;\Longrightarrow\;\ln y=\alpha\,\ln x+\ln k .
\]
On pose $Z=\ln y$ (et éventuellement $T=\ln x$), on ajuste $Z$ par une droite, puis on
\textbf{revient} à $y$.
\end{aretenir}

\begin{exemple}
Un nuage $(x_i,y_i)$ semble exponentiel. On pose $Z_i=\ln y_i$ et l'ajustement de $Z$ en
$x$ donne $Z=0{,}3\,x+1{,}1$. Par identification avec $\ln y=mx+\ln k$ : $m=0{,}3$ et
$k=e^{1{,}1}\approx3{,}00$. Le modèle reconstitué est $y\approx3\,e^{0{,}3x}$.
\end{exemple}

\section{L'essentiel à retenir}

\begin{synthese}
\textbf{Une variable — position.} Moyenne $\bar x=\dfrac1N\sum n_i x_i=\sum f_i x_i$.
Médiane $M_e$ : valeur centrale de la série \emph{ordonnée} (robuste). Mode = valeur la
plus fréquente. Quartiles $Q_1,Q_2,Q_3$ ; écart interquartile $Q_3-Q_1$.\\[4pt]
\textbf{Une variable — dispersion.} Variance $V(X)=\dfrac1N\sum n_i(x_i-\bar x)^2
=\overline{x^2}-\bar x^2$ (König). Écart-type $\sigma_X=\sqrt{V(X)}$, même unité que $X$.\\[4pt]
\textbf{Transformation affine.} $y=ax+b\Rightarrow \bar y=a\bar x+b$, $V(Y)=a^2V(X)$,
$\sigma_Y=|a|\sigma_X$.\\[4pt]
\textbf{Deux variables.} Point moyen $G(\bar x;\bar y)$.
$\mathrm{cov}(X,Y)=\overline{xy}-\bar x\,\bar y$.
Corrélation $r=\dfrac{\mathrm{cov}(X,Y)}{\sigma_X\sigma_Y}\in[-1,1]$.\\[4pt]
\textbf{Régression (moindres carrés).} $y$ en $x$ : $a=\dfrac{\mathrm{cov}(X,Y)}{V(X)}$,
$b=\bar y-a\bar x$. $x$ en $y$ : $a'=\dfrac{\mathrm{cov}(X,Y)}{V(Y)}$, $b'=\bar x-a'\bar y$.
Les deux droites passent par $G$ et $r^2=a\,a'$.\\[4pt]
\textbf{Prévision.} On remplace dans la droite de $y$ en $x$ (interpolation /
extrapolation). Ajustement justifié si $|r|$ proche de $1$ ($|r|\ge0{,}87$).\\[4pt]
\textbf{Changement de variable.} $y=ke^{mx}\Rightarrow Z=\ln y=mx+\ln k$ ;
$y=kx^\alpha\Rightarrow \ln y=\alpha\ln x+\ln k$. On ajuste, puis on revient à $y$.\\[4pt]
\textbf{Pièges.} König donne parfois $V<0$ si un carré est oublié ; corrélation $\ne$
causalité ; histogramme = aire (pas hauteur) si amplitudes inégales ; pour prévoir $y$,
toujours la droite \emph{de $y$ en $x$}.
\end{synthese}

\section{Méthodes \& savoir-faire}

\methode{Calculer moyenne, variance et écart-type d'une série pondérée}
Dresser un tableau $x_i$, $n_i$, $n_ix_i$, $n_ix_i^2$ ; sommer ; appliquer
$\bar x=\frac1N\sum n_ix_i$ et König $V=\frac1N\sum n_ix_i^2-\bar x^2$.
\begin{exemple}
Notes $x_i$ : $8,12,14$ d'effectifs $n_i$ : $5,8,2$ ($N=15$).
$\sum n_ix_i=40+96+28=164\Rightarrow\bar x=\dfrac{164}{15}\approx10{,}93$.
$\sum n_ix_i^2=5\!\cdot\!64+8\!\cdot\!144+2\!\cdot\!196=320+1152+392=1864$,
$V=\dfrac{1864}{15}-\bar x^2\approx124{,}27-119{,}51=4{,}76$, $\sigma\approx2{,}18$.
\end{exemple}

\methode{Déterminer médiane et quartiles d'une série}
Ordonner ; repérer le rang $\frac{N+1}2$ (médiane), $\frac N4$ et $\frac{3N}4$
(quartiles), avec la règle d'arrondi à l'entier supérieur si non entier.
\begin{exemple}
$N=9$ valeurs ordonnées : $2,4,4,5,7,9,10,12,15$. Médiane = valeur de rang
$\frac{9+1}2=5$, soit $M_e=7$. $Q_1$ : rang $\frac94=2{,}25\to 3^{\text{e}}$ valeur $=4$ ;
$Q_3$ : rang $\frac{27}4=6{,}75\to 7^{\text{e}}$ valeur $=10$.
\end{exemple}

\methode{Représenter un nuage et placer le point moyen $G$}
Reporter les points $M_i(x_i\,;\,y_i)$ ; calculer $\bar x$ et $\bar y$ ; placer $G(\bar x\,;\,\bar y)$.
\begin{exemple}
Pour la série $\;(1;2)\;(2;3)\;(3;5)\;(4;6)$ : $\bar x=\dfrac{1+2+3+4}{4}=2{,}5$ et
$\bar y=\dfrac{2+3+5+6}{4}=4$. Le point moyen est $G(2{,}5\,;\,4)$.
\end{exemple}

\methode{Calculer covariance et coefficient de corrélation}
Dresser un tableau de $x_i$, $y_i$, $x_i^2$, $y_i^2$, $x_iy_i$. Puis appliquer König :
$\mathrm{cov}=\overline{xy}-\bar x\bar y$, $V(X)=\overline{x^2}-\bar x^2$, $V(Y)=\overline{y^2}-\bar y^2$, et $r=\dfrac{\mathrm{cov}}{\sigma_X\sigma_Y}$.
\begin{exemple}
Série : $\;(1;3)\;(2;4)\;(3;6)\;(4;7)$ ($n=4$).
$\bar x=2{,}5$, $\bar y=5$.
$\sum x_iy_i=1\!\cdot\!3+2\!\cdot\!4+3\!\cdot\!6+4\!\cdot\!7=3+8+18+28=57$, donc
$\overline{xy}=14{,}25$ et $\mathrm{cov}=14{,}25-2{,}5\times5=1{,}75$.
$\sum x_i^2=1+4+9+16=30\Rightarrow V(X)=7{,}5-6{,}25=1{,}25$.
$\sum y_i^2=9+16+36+49=110\Rightarrow V(Y)=27{,}5-25=2{,}5$.
Ainsi $r=\dfrac{1{,}75}{\sqrt{1{,}25\times2{,}5}}=\dfrac{1{,}75}{\sqrt{3{,}125}}\approx\dfrac{1{,}75}{1{,}7678}\approx0{,}990$ : corrélation très forte.
\end{exemple}

\methode{Déterminer la droite de régression de $y$ en $x$}
Calculer $a=\dfrac{\mathrm{cov}(X,Y)}{V(X)}$ puis $b=\bar y-a\bar x$ ; écrire $y=ax+b$.
\begin{exemple}
Avec l'exemple précédent : $a=\dfrac{1{,}75}{1{,}25}=1{,}4$ et $b=5-1{,}4\times2{,}5=5-3{,}5=1{,}5$.
La droite est $\;y=1{,}4\,x+1{,}5\;$ ; elle passe bien par $G(2{,}5\,;\,5)$ car $1{,}4\times2{,}5+1{,}5=5$.
\end{exemple}

\methode{Estimer et prévoir par interpolation/extrapolation}
Remplacer la valeur connue dans l'équation de la droite de $y$ en $x$ pour estimer la valeur
inconnue de $y$ (et inversement avec la droite de $x$ en $y$).
\begin{exemple}
Avec $y=1{,}4x+1{,}5$, la prévision pour $x=6$ est $y=1{,}4\times6+1{,}5=8{,}4+1{,}5=9{,}9$.
\end{exemple}

\methode{Ajuster par changement de variable (modèle exponentiel)}
Poser $Z=\ln y$, ajuster $Z$ par une droite $Z=mx+p$ (moindres carrés sur $(x_i,Z_i)$), puis
revenir : $y=e^{p}\,e^{mx}$, c'est-à-dire $k=e^{p}$ et le taux $m$.
\begin{exemple}
Si l'ajustement de $Z=\ln y$ donne $Z=0{,}2\,x+0{,}7$, alors $m=0{,}2$ et $k=e^{0{,}7}\approx2{,}01$ :
le modèle est $y\approx2{,}01\,e^{0{,}2x}$. Pour $x=10$ : $y\approx2{,}01\,e^{2}\approx2{,}01\times7{,}389\approx14{,}9$.
\end{exemple}

\methode{Utiliser la relation $r^2=a\,a'$ et le passage par $G$}
Contrôler ses calculs : $r^2=a\,a'$, et les deux droites se coupent en $G$. Si l'une de ces
vérifications échoue, c'est qu'il y a une erreur de calcul.
\begin{exemple}
Si $a=1{,}6$ et $a'=0{,}5$ avec un nuage croissant, alors $r^2=0{,}8$, soit $r=\sqrt{0{,}8}\approx0{,}894>0$.
Cohérent avec une covariance positive.
\end{exemple}

\methode{Exploiter l'effet d'une transformation affine sur $\bar x$, $V$, $\sigma$}
Pour simplifier des calculs, poser $u_i=\dfrac{x_i-x_0}{h}$ (changement d'origine $x_0$ et
d'échelle $h$) : alors $\bar x=x_0+h\,\bar u$ et $\sigma_X=h\,\sigma_U$.
\begin{exemple}
Données $1000,1010,1020,1030$. Avec $u=\frac{x-1000}{10}$ : $u=0,1,2,3$, $\bar u=1{,}5$,
donc $\bar x=1000+10\times1{,}5=1015$ ; $V(U)=\overline{u^2}-\bar u^2=3{,}5-2{,}25=1{,}25$,
$\sigma_U\approx1{,}118$, d'où $\sigma_X=10\,\sigma_U\approx11{,}18$.
\end{exemple}
