\section{Sujet n\textsuperscript{o}\,1}
\noindent{\small\itshape Épreuve de Mathématiques — Baccalauréat série C · Durée : 4\,h · Coefficient : 7.
L'épreuve comporte deux parties : l'\textbf{évaluation des ressources} et l'\textbf{évaluation
des compétences}. La clarté de la rédaction et le soin sont notés.}
\par\smallskip{\color{onbo}\rule{\linewidth}{1pt}}

\subsection*{Partie A — Évaluation des ressources \hfill (15,5 points)}

\noindent\textbf{Exercice 1.} \hfill\textit{(4,5 points)}\\
On considère dans $\C$ l'équation $(E):\ z^2-2z+4=0$.
\begin{enumerate}[label=\arabic*)]
\item Résoudre $(E)$ ; on note $z_1$ la solution de partie imaginaire positive et $z_2$ l'autre. \hfill\textit{(1 pt)}
\item Écrire $z_1$ et $z_2$ sous forme trigonométrique, puis calculer $z_1^{\,3}$. \hfill\textit{(1,5 pt)}
\item Le plan est muni d'un repère orthonormé direct. $A$, $B$, $C$ ont pour affixes
$z_1$, $z_2$, $2$. Démontrer que $ABC$ est isocèle et préciser le sommet principal. \hfill\textit{(2 pts)}
\end{enumerate}

\smallskip
\noindent\textbf{Exercice 2.} \hfill\textit{(5 points)}\\
Une urne contient $4$ boules rouges et $6$ boules vertes indiscernables au toucher. On
tire simultanément $3$ boules. Soit $X$ le nombre de boules rouges obtenues.
\begin{enumerate}[label=\arabic*)]
\item Justifier qu'il y a $120$ tirages possibles. \hfill\textit{(1 pt)}
\item Déterminer la loi de probabilité de $X$. \hfill\textit{(2 pts)}
\item Calculer $E(X)$ et la probabilité de tirer au moins une boule rouge. \hfill\textit{(2 pts)}
\end{enumerate}

\smallskip
\noindent\textbf{Exercice 3.} \hfill\textit{(6 points)}\\
Soit $f$ définie sur $\R$ par $f(x)=(x+1)\mathrm{e}^{-x}$, de courbe $(\mathcal{C})$.
\begin{enumerate}[label=\arabic*)]
\item Déterminer les limites de $f$ en $-\infty$ et $+\infty$ ; interpréter graphiquement celle en $+\infty$. \hfill\textit{(1,5 pt)}
\item Montrer que $f'(x)=-x\,\mathrm{e}^{-x}$ et dresser le tableau de variations de $f$. \hfill\textit{(1,5 pt)}
\item À l'aide d'une intégration par parties, montrer que $\displaystyle\int_0^{\lambda}f(x)\,\dd x=2-(\lambda+2)\mathrm{e}^{-\lambda}$,
puis calculer sa limite en $+\infty$. \hfill\textit{(3 pts)}
\end{enumerate}

\subsection*{Partie B — Évaluation des compétences \hfill (4,5 points)}

\noindent\textit{Monsieur Abena exploite une plantation de cacao près d'Obala. Pour la
nouvelle campagne, il fait trois projets et te sollicite, en tant qu'élève de Terminale~C,
pour l'aider à décider.}

\smallskip
\noindent\textbf{Tâche 1.} Il veut clôturer une parcelle \emph{rectangulaire} adossée à un
mur déjà existant (le côté le long du mur n'est pas clôturé). Il dispose de \SI{200}{\metre}
de grillage. Détermine les dimensions de la parcelle d'\emph{aire maximale} qu'il peut clôturer,
et donne cette aire. \hfill\textit{(1,5 pt)}

\smallskip
\noindent\textbf{Tâche 2.} Sa production de cacao est de \SI{1,2}{\tonne} la première année et
augmente de \SI{8}{\percent} chaque année. Détermine l'année à partir de laquelle la
production annuelle dépassera \SI{2}{\tonne}. \hfill\textit{(1,5 pt)}

\smallskip
\noindent\textbf{Tâche 3.} À la récolte, \SI{5}{\percent} des sacs sont mal séchés. Un acheteur
prélève au hasard $4$ sacs. M. Abena veut connaître le risque qu'\emph{au moins un} sac
prélevé soit mal séché. Détermine cette probabilité. \hfill\textit{(1,5 pt)}

\par\smallskip{\footnotesize\itshape Présentation et cohérence de la copie : 0,5 point. La situation
est évaluée selon les critères : pertinence de la production, utilisation correcte des outils
mathématiques, cohérence de la production.}

\noindent\textbf{Partie A — Exercice 1.}
\begin{enumerate}[label=\arabic*)]
\item $\Delta=4-16=-12=(2i\sqrt3)^2$, donc $z=1\pm i\sqrt3$ : $z_1=1+i\sqrt3$, $z_2=1-i\sqrt3$.
\item $|z_1|=2$, $\arg z_1=\frac\pi3$ : $z_1=2(\cos\frac\pi3+i\sin\frac\pi3)$ et $z_2=2(\cos\frac\pi3-i\sin\frac\pi3)$.
$z_1^{\,3}=2^3\mathrm{e}^{i\pi}=-8$.
\item $AB=|z_1-z_2|=2\sqrt3$, $AC=|z_1-2|=|-1+i\sqrt3|=2$, $BC=|z_2-2|=2$. Donc $AC=BC$ :
$ABC$ est isocèle de sommet principal $C$.
\end{enumerate}

\noindent\textbf{Exercice 2.}
\begin{enumerate}[label=\arabic*)]
\item Choix de $3$ boules parmi $10$ : $\binom{10}{3}=120$.
\item $P(X{=}k)=\dfrac{\binom4k\binom6{3-k}}{120}$ : $P(0)=\frac16$, $P(1)=\frac12$, $P(2)=\frac3{10}$, $P(3)=\frac1{30}$.
\item $E(X)=\frac12+\frac35+\frac1{10}=\frac65=1{,}2$ ; $P(X\ge1)=1-\frac16=\frac56$.
\end{enumerate}

\noindent\textbf{Exercice 3.}
1) En $+\infty$, $f(x)\to0^+$ (asymptote $y=0$) ; en $-\infty$, $f(x)\to-\infty$.
2) $f'(x)=\mathrm{e}^{-x}-(x+1)\mathrm{e}^{-x}=-x\,\mathrm{e}^{-x}$ ; $f$ croît sur $]-\infty,0]$, décroît sur $[0,+\infty[$, maximum $f(0)=1$.
3) IPP avec $u=x+1$, $v'=\mathrm{e}^{-x}$ : $\int_0^\lambda f=\big[-(x+2)\mathrm{e}^{-x}\big]_0^\lambda=2-(\lambda+2)\mathrm{e}^{-\lambda}$, de limite $2$ en $+\infty$.

\smallskip
\noindent\textbf{Partie B — Situation-problème.}
\emph{Tâche 1 (optimisation).} Soit $x$ la largeur (les deux côtés perpendiculaires au mur) et
$y$ la longueur clôturée. Le grillage couvre trois côtés : $2x+y=200$, donc $y=200-2x$.
L'aire $A(x)=x(200-2x)=200x-2x^2$ ; $A'(x)=200-4x$ s'annule en $x=50$. L'aire est maximale
pour $x=\SI{50}{\metre}$ et $y=\SI{100}{\metre}$, soit $A=\SI{5000}{\metre\squared}$.

\emph{Tâche 2 (suite géométrique).} La production de l'année $n$ est $u_n=1{,}2\times1{,}08^{\,n-1}$.
On cherche $n$ tel que $u_n>2$, soit $1{,}08^{\,n-1}>\frac{2}{1{,}2}=\frac53$. En passant au logarithme :
$n-1>\frac{\ln(5/3)}{\ln 1{,}08}\approx\frac{0{,}511}{0{,}077}\approx6{,}6$, d'où $n\ge 8$.
C'est donc à partir de la \textbf{8\textsuperscript{e} année} (production $\approx\SI{2,06}{\tonne}$).

\emph{Tâche 3 (probabilité).} Chaque sac est mal séché avec la probabilité $0{,}05$,
indépendamment. $P(\text{au moins un})=1-P(\text{aucun})=1-(0{,}95)^4\approx1-0{,}815=\mathbf{0{,}185}$,
soit environ \SI{18,5}{\percent}.
