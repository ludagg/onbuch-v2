\chapter{Limites et continuité}

\noindent\textit{La notion de limite décrit le comportement d'une fonction au voisinage d'un
point ou de l'infini ; la continuité traduit l'idée d'une courbe « tracée sans lever le
crayon ». Ces deux notions ouvrent l'analyse de Terminale et fondent l'étude des fonctions.}
\vspace{8pt}

% ═══════════════════════════════════════════════════════════════
\section{Limites d'une fonction}

\begin{definition}[Limite finie ou infinie]
Soit $f$ une fonction et $a$ un réel (ou $a=\pm\infty$).
\begin{itemize}
\item $\displaystyle\lim_{x\to a}f(x)=\ell$ : $f(x)$ devient aussi proche que l'on veut de $\ell$ lorsque $x$ se rapproche de $a$.
\item $\displaystyle\lim_{x\to a}f(x)=+\infty$ : $f(x)$ dépasse tout nombre fixé lorsque $x$ se rapproche de $a$.
\end{itemize}
\end{definition}

\begin{propriete}[Limites de référence]
\[
\lim_{x\to+\infty}x^{n}=+\infty,\quad
\lim_{x\to+\infty}\frac1{x^{n}}=0,\quad
\lim_{x\to+\infty}\sqrt{x}=+\infty,\quad
\lim_{x\to0^{+}}\frac1x=+\infty,\quad
\lim_{x\to0^{-}}\frac1x=-\infty.
\]
\end{propriete}

\begin{theoreme}[Opérations et formes indéterminées]
La limite d'une somme, d'un produit ou d'un quotient se calcule avec les limites, \emph{sauf} dans les quatre \textbf{formes indéterminées} :
\[
\boxed{\;\infty-\infty\;}\qquad\boxed{\;0\times\infty\;}\qquad\boxed{\;\tfrac{\infty}{\infty}\;}\qquad\boxed{\;\tfrac{0}{0}\;}
\]
\end{theoreme}

\begin{remarque}
En $\pm\infty$, une fonction \textbf{polynôme} a la limite de son terme de plus haut degré ; une fonction \textbf{rationnelle}, celle du quotient des termes de plus haut degré.
\end{remarque}

\begin{theoreme}[Limite d'une composée]
Si $\displaystyle\lim_{x\to a}u(x)=b$ et $\displaystyle\lim_{X\to b}g(X)=\ell$, alors $\displaystyle\lim_{x\to a}g\big(u(x)\big)=\ell$.
\end{theoreme}

\section{Continuité}

\begin{definition}[Continuité]
$f$ est \textbf{continue en} $a$ si $f$ est définie en $a$ et $\displaystyle\lim_{x\to a}f(x)=f(a)$. Elle est continue sur un intervalle $I$ si elle l'est en chaque point de $I$.
\end{definition}

\begin{propriete}[Continuité des fonctions usuelles]
Les fonctions polynômes, rationnelles, racine carrée, valeur absolue, ainsi que leurs sommes, produits, quotients (où ils existent) et composées, sont continues sur tout intervalle de leur domaine.
\end{propriete}

\begin{definition}[Prolongement par continuité]
Si $f$ n'est pas définie en $a$ mais admet une limite finie $\ell$ en $a$, on la \textbf{prolonge par continuité} en posant $\tilde f(a)=\ell$.
\end{definition}

\begin{theoreme}[Théorème des valeurs intermédiaires (TVI)]
Si $f$ est continue sur $[a,b]$, alors pour tout réel $k$ compris entre $f(a)$ et $f(b)$, l'équation $f(x)=k$ admet \textbf{au moins une} solution dans $[a,b]$.
\end{theoreme}

\begin{propriete}[Corollaire : cas strictement monotone]
Si de plus $f$ est \textbf{strictement monotone} sur $[a,b]$, l'équation $f(x)=k$ admet une \textbf{unique} solution dans $[a,b]$.
\end{propriete}

\begin{aretenir}
Pour montrer que $f(x)=0$ a une unique solution sur $[a,b]$ : (1) $f$ \textbf{continue} ; (2) $f$ \textbf{strictement monotone} ; (3) $f(a)\,f(b)<0$.
\end{aretenir}

\section{Méthodes \& savoir-faire}

\methode{Lever une forme indéterminée}
Factoriser par le terme dominant (polynômes/rationnelles), utiliser l'expression conjuguée (racines) ou factoriser (forme $\tfrac00$).
\begin{exemple}
$\displaystyle\lim_{x\to+\infty}\big(\sqrt{x^{2}+x}-x\big)=\lim_{x\to+\infty}\frac{x}{\sqrt{x^{2}+x}+x}=\lim_{x\to+\infty}\frac{1}{\sqrt{1+\frac1x}+1}=\frac12.$
\end{exemple}

\methode{Étudier la continuité ou prolonger}
Comparer $\displaystyle\lim_{x\to a}f$ et $f(a)$ ; prolonger si la limite est finie.
\begin{exemple}
$f(x)=\dfrac{x^{2}-1}{x-1}=x+1$ pour $x\neq1$ : $\displaystyle\lim_{x\to1}f=2$, on pose $\tilde f(1)=2$.
\end{exemple}

\methode{Appliquer le TVI (existence et unicité d'une solution)}
Vérifier continuité, signe aux bornes, stricte monotonie ; puis encadrer par balayage.
\begin{exemple}
$f(x)=x^{3}+x-1$ : $f'(x)=3x^{2}+1>0$, $f(0)=-1$, $f(1)=1$ : unique solution $\alpha\in\,]0,1[$ ; $f(0{,}6)<0<f(0{,}7)$ donne $0{,}6<\alpha<0{,}7$.
\end{exemple}

\section{Exercices}

\rubrique{Calculs de limites}
\exo{}\diff{1} Calculer $\displaystyle\lim_{x\to+\infty}\frac{3x^{2}-x+2}{x^{2}+5}$ et $\displaystyle\lim_{x\to2}\frac{x^{2}-4}{x-2}$.

\exo{}\diff{2} Calculer $\displaystyle\lim_{x\to+\infty}\big(\sqrt{x^{2}+3x}-x\big)$ et $\displaystyle\lim_{x\to1}\frac{\sqrt x-1}{x-1}$.

\rubrique{Continuité}
\exo{}\diff{1} La fonction $f(x)=\dfrac{x^{2}-9}{x-3}$ admet-elle un prolongement par continuité en $3$ ?

\exo{}\diff{2} Déterminer $a$ pour que $f(x)=\begin{cases}x^{2}+1&x\le1\\ ax+2&x>1\end{cases}$ soit continue en $1$.

\rubrique{Théorème des valeurs intermédiaires — Problème type Baccalauréat}
\exo{}\diff{2} Soit $f(x)=x^{3}-3x+1$. Montrer que $f(x)=0$ admet une solution $\alpha\in\,]1,2[$ et l'encadrer à $0{,}1$ près.

\exo{}\diff{3} Soit $f(x)=\dfrac{x-1}{x+1}$ sur $[0,+\infty[$.
\begin{enumerate}[label=\arabic*)]
\item Montrer que $f(x)=1-\dfrac{2}{x+1}$ et en déduire son sens de variation.
\item Calculer $\displaystyle\lim_{x\to+\infty}f$ et interpréter.
\item Montrer que $f(x)=\frac12$ a une unique solution sur $[0,+\infty[$ et la déterminer.
\end{enumerate}

\section{Corrigés}

\corrige{de l'exercice 1}
$\dfrac{3x^{2}-x+2}{x^{2}+5}\to\dfrac{3x^{2}}{x^{2}}=3$ ; $\dfrac{x^{2}-4}{x-2}=x+2\to4$.

\corrige{de l'exercice 2}
$\sqrt{x^{2}+3x}-x=\dfrac{3x}{\sqrt{x^{2}+3x}+x}\to\dfrac32$ ; $\dfrac{\sqrt x-1}{x-1}=\dfrac1{\sqrt x+1}\to\dfrac12$.

\corrige{de l'exercice 3}
Pour $x\neq3$, $f(x)=x+3\to6$ : limite finie, donc $f$ se prolonge en posant $\tilde f(3)=6$.

\corrige{de l'exercice 4}
À gauche $\lim=2$, à droite $\lim=a+2$ ; continuité $\Leftrightarrow a+2=2\Leftrightarrow a=0$.

\corrige{de l'exercice 5}
$f$ polynôme (continue), $f(1)=-1<0$, $f(2)=3>0$ : par le TVI, $\alpha\in\,]1,2[$. $f(1{,}5)=-0{,}125<0$, $f(1{,}6)=0{,}296>0$ : $1{,}5<\alpha<1{,}6$.

\corrige{du problème}
1) $1-\dfrac{2}{x+1}=\dfrac{x-1}{x+1}=f(x)$ ; comme $x\mapsto\dfrac2{x+1}$ décroît, $f$ \textbf{croît} sur $[0,+\infty[$.\;
2) $\displaystyle\lim_{x\to+\infty}f=1$ : asymptote horizontale $y=1$.\;
3) $f$ continue, strictement croissante, $f(0)=-1$, $\lim_{+\infty}f=1$, et $\frac12\in\,]-1,1[$ : unique solution. $\dfrac{x-1}{x+1}=\dfrac12\Leftrightarrow 2x-2=x+1\Leftrightarrow x=3$.
