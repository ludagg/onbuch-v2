\chapter{Algèbre linéaire : espaces vectoriels et applications linéaires}

\noindent\textit{Les vecteurs du plan et de l'espace que tu manipules depuis la seconde
obéissent à des règles communes : on les additionne, on les multiplie par un nombre, on
repère un point par ses coordonnées. En dégageant ces règles, on obtient le \textbf{calcul
vectoriel} et la notion d'\textbf{espace vectoriel}, un cadre unique qui décrit aussi bien
$\R^2$, $\R^3$ que beaucoup d'autres ensembles. Ce chapitre, spécificité de la série C,
rassemble tout l'outillage : combinaisons linéaires, colinéarité et coplanarité, bases et
repères, coordonnées, \textbf{déterminant} de deux vecteurs du plan, applications
géométriques (alignement, parallélisme) et \textbf{systèmes d'équations linéaires}. Il prépare
directement le calcul matriciel et éclaire toute la géométrie analytique du Baccalauréat.}
\vspace{8pt}

\section{Vecteurs du plan et de l'espace}

\begin{definition}[Vecteur]
À tout couple de points $(A,B)$ on associe le \textbf{vecteur} $\vec{AB}$, caractérisé par
sa \textbf{direction} (celle de la droite $(AB)$), son \textbf{sens} (de $A$ vers $B$) et sa
\textbf{norme} (ou longueur) $\|\vec{AB}\|=AB$. Deux bipoints $(A,B)$ et $(C,D)$ représentent
le \textbf{même} vecteur si et seulement si $ABDC$ est un parallélogramme, c'est-à-dire si
$\vec{AB}=\vec{CD}$. Le \textbf{vecteur nul} $\vec 0=\vec{AA}$ est de norme $0$.
\end{definition}

\begin{propriete}[Relation de Chasles et opérations]
Pour tous points $A,B,C$ du plan ou de l'espace :
\[
\vec{AB}+\vec{BC}=\vec{AC}\qquad\text{(relation de Chasles)},\qquad \vec{AB}=-\vec{BA}.
\]
L'addition de deux vecteurs se construit « bout à bout » (Chasles) ou par la
\textbf{règle du parallélogramme}. La multiplication par un réel $\lambda$ donne un vecteur
$\lambda\vec u$ de même direction que $\vec u$, de norme $|\lambda|\,\|\vec u\|$, de même sens
si $\lambda>0$, de sens opposé si $\lambda<0$.
\end{propriete}

\begin{remarque}
Un vecteur n'est pas « attaché » à un point : il représente seulement une translation. On peut
le « déplacer » librement tant qu'on conserve direction, sens et norme. C'est ce qui permet de
ramener tout vecteur à une origine commune $O$ : $\vec u=\vec{OM}$ pour un unique point $M$.
\end{remarque}

\section{Espaces vectoriels réels}

\begin{definition}[Espace vectoriel réel]
Un ensemble $E$ non vide est un \textbf{espace vectoriel sur} $\R$ (ses éléments sont appelés
\textbf{vecteurs}) lorsqu'il est muni :
\begin{itemize}
\item d'une \textbf{addition} $+$ qui à deux vecteurs $\vec u,\vec v$ associe $\vec u+\vec v\in E$ ;
\item d'une \textbf{multiplication par un réel} (un \textbf{scalaire}) qui à $\lambda\in\R$ et $\vec u\in E$ associe $\lambda\vec u\in E$ ;
\end{itemize}
de sorte que, pour tous $\vec u,\vec v,\vec w\in E$ et tous $\lambda,\mu\in\R$ :
\[
\begin{array}{llll}
\vec u+\vec v=\vec v+\vec u, & (\vec u+\vec v)+\vec w=\vec u+(\vec v+\vec w), & \vec u+\vec 0=\vec u, & \vec u+(-\vec u)=\vec 0,\\[3pt]
\lambda(\vec u+\vec v)=\lambda\vec u+\lambda\vec v, & (\lambda+\mu)\vec u=\lambda\vec u+\mu\vec u, & \lambda(\mu\vec u)=(\lambda\mu)\vec u, & 1\cdot\vec u=\vec u.
\end{array}
\]
Le vecteur $\vec 0$ est le \textbf{vecteur nul}.
\end{definition}

\begin{exemple}
$\R^2=\{(x,y)\mid x,y\in\R\}$ est un espace vectoriel pour
$(x,y)+(x',y')=(x+x',y+y')$ et $\lambda(x,y)=(\lambda x,\lambda y)$, de vecteur nul $(0,0)$.
De même $\R^3=\{(x,y,z)\}$. Plus généralement $\R^n$ est un espace vectoriel.
\end{exemple}

\begin{remarque}
L'ensemble des vecteurs du plan (ou de l'espace) géométrique est un espace vectoriel : c'est
le modèle que l'on visualise. On note souvent un vecteur de $\R^2$ aussi bien $(x,y)$ que
$\vec u$. Le plan vectoriel a \textbf{dimension $2$}, l'espace vectoriel \textbf{dimension $3$}.
\end{remarque}

\begin{propriete}[Règles de calcul]
Dans tout espace vectoriel $E$, pour tout $\vec u\in E$ et tout $\lambda\in\R$ :
\[
0\cdot\vec u=\vec 0,\qquad \lambda\cdot\vec 0=\vec 0,\qquad (-1)\vec u=-\vec u,\qquad
\lambda\vec u=\vec 0\ \Longleftrightarrow\ \lambda=0\ \text{ou}\ \vec u=\vec 0.
\]
\end{propriete}

\section{Sous-espaces vectoriels}

\begin{definition}[Sous-espace vectoriel]
Une partie $F$ de $E$ est un \textbf{sous-espace vectoriel} (s.e.v.) de $E$ si :
\begin{enumerate}[label=\arabic*)]
\item $F$ contient le vecteur nul : $\vec 0\in F$ (donc $F\neq\varnothing$) ;
\item $F$ est \textbf{stable par addition} : pour tous $\vec u,\vec v\in F$, $\vec u+\vec v\in F$ ;
\item $F$ est \textbf{stable par multiplication par un scalaire} : pour tout $\lambda\in\R$ et tout $\vec u\in F$, $\lambda\vec u\in F$.
\end{enumerate}
\end{definition}

\begin{theoreme}[Caractérisation pratique]
$F\subset E$ est un sous-espace vectoriel de $E$ si et seulement si :
\[
\vec 0\in F\qquad\text{et}\qquad \forall\,\vec u,\vec v\in F,\ \forall\,\lambda,\mu\in\R,\quad \lambda\vec u+\mu\vec v\in F.
\]
Un sous-espace vectoriel est lui-même un espace vectoriel : c'est l'intérêt de cette notion.
\end{theoreme}

\begin{exemple}
Dans $\R^2$, $F=\{(x,y)\mid y=2x\}$ est un s.e.v. : $(0,0)\in F$ ; si $\vec u=(x,2x)$ et
$\vec v=(x',2x')$, alors $\lambda\vec u+\mu\vec v=(\lambda x+\mu x',\,2(\lambda x+\mu x'))\in F$.
Géométriquement, $F$ est une \textbf{droite vectorielle} (passant par l'origine).
\end{exemple}

\begin{remarque}
$\{\vec 0\}$ et $E$ tout entier sont toujours des sous-espaces vectoriels de $E$. En revanche,
une droite \emph{ne passant pas} par l'origine n'est \textbf{pas} un s.e.v. (elle ne contient
pas $\vec 0$). Les s.e.v. de $\R^2$ sont : $\{\vec 0\}$, les droites vectorielles, et $\R^2$.
Ceux de $\R^3$ : $\{\vec 0\}$, les droites vectorielles, les plans vectoriels, et $\R^3$.
\end{remarque}

\section{Combinaisons linéaires, familles génératrices}

\begin{definition}[Combinaison linéaire]
Une \textbf{combinaison linéaire} des vecteurs $\vec{u_1},\dots,\vec{u_p}$ est tout vecteur de la forme
\[
\lambda_1\vec{u_1}+\lambda_2\vec{u_2}+\dots+\lambda_p\vec{u_p},\qquad (\lambda_1,\dots,\lambda_p)\in\R^p.
\]
Les réels $\lambda_i$ sont les \textbf{coefficients} de la combinaison.
\end{definition}

\begin{definition}[Famille génératrice]
La famille $(\vec{u_1},\dots,\vec{u_p})$ est \textbf{génératrice} de $E$ si \emph{tout} vecteur
de $E$ s'écrit comme combinaison linéaire de ces vecteurs. On note alors
$E=\mathrm{Vect}(\vec{u_1},\dots,\vec{u_p})$.
\end{definition}

\begin{exemple}
Dans $\R^2$, $\vec i=(1,0)$ et $\vec j=(0,1)$ engendrent $\R^2$ : tout $(x,y)=x\vec i+y\vec j$.
La famille $(\vec i,\vec j)$ est génératrice. De même, on vérifie que $\vec u=(2,1)$ est
combinaison linéaire de $\vec a=(1,1)$ et $\vec b=(0,1)$ : $\vec u=2\vec a-\vec b$.
\end{exemple}

\begin{remarque}
$\mathrm{Vect}(\vec{u_1},\dots,\vec{u_p})$ est toujours un sous-espace vectoriel de $E$
(le « plus petit » contenant ces vecteurs). $\mathrm{Vect}(\vec u)$ est la droite vectorielle
dirigée par $\vec u$ (si $\vec u\neq\vec 0$) ; $\mathrm{Vect}(\vec u,\vec v)$ est le plan
vectoriel engendré par $\vec u$ et $\vec v$ (si $\vec u,\vec v$ non colinéaires).
\end{remarque}

\section{Colinéarité, familles libres et liées}

\begin{definition}[Vecteurs colinéaires]
Deux vecteurs $\vec u$ et $\vec v$ sont \textbf{colinéaires} s'il existe un réel $\lambda$ tel que
$\vec v=\lambda\vec u$ (ou $\vec u=\vec 0$). Géométriquement, ils ont la \textbf{même direction} :
les points qui les portent sont alignés ou les droites sont parallèles.
\end{definition}

\begin{definition}[Famille libre, famille liée]
La famille $(\vec{u_1},\dots,\vec{u_p})$ est \textbf{libre} (les vecteurs sont \emph{linéairement indépendants}) si la seule combinaison linéaire nulle est triviale :
\[
\lambda_1\vec{u_1}+\dots+\lambda_p\vec{u_p}=\vec 0\ \Longrightarrow\ \lambda_1=\dots=\lambda_p=0.
\]
Sinon la famille est \textbf{liée} : l'un des vecteurs est combinaison linéaire des autres.
\end{definition}

\begin{propriete}[Cas de deux vecteurs]
Deux vecteurs $\vec u,\vec v$ forment une famille \textbf{libre} si et seulement s'ils ne sont
pas colinéaires. Ils sont \textbf{liés} si et seulement s'ils sont colinéaires
($\vec v=\lambda\vec u$ pour un certain $\lambda$, ou $\vec u=\vec 0$).
\end{propriete}

\begin{exemple}
$\vec u=(1,2)$ et $\vec v=(2,4)$ sont liés car $\vec v=2\vec u$. En revanche $\vec u=(1,2)$ et
$\vec w=(3,1)$ sont libres : ils ne sont pas colinéaires ($1\times1-2\times3=-5\neq0$).
\end{exemple}

\begin{aretenir}
Dans $\R^2$ : deux vecteurs sont \textbf{colinéaires} $\Leftrightarrow$ leur déterminant est nul
(voir plus bas). Dans $\R^3$, trois vecteurs sont \textbf{coplanaires} (liés) si l'un est
combinaison linéaire des deux autres ; sinon ils forment une famille libre, donc une base.
\end{aretenir}

\section{Coplanarité dans l'espace}

\begin{definition}[Vecteurs coplanaires]
Trois vecteurs $\vec u,\vec v,\vec w$ de l'espace sont \textbf{coplanaires} s'il existe des points
$A,B,C,D$ d'un même plan tels que $\vec u=\vec{AB}$, $\vec v=\vec{AC}$, $\vec w=\vec{AD}$.
De façon équivalente : la famille $(\vec u,\vec v,\vec w)$ est \textbf{liée}.
\end{definition}

\begin{propriete}[Caractérisation de la coplanarité]
Soient $\vec u,\vec v$ deux vecteurs \textbf{non colinéaires} de l'espace. Un vecteur $\vec w$ est
coplanaire avec $\vec u$ et $\vec v$ si et seulement s'il existe deux réels $a,b$ tels que
\[
\vec w=a\,\vec u+b\,\vec v.
\]
On dit que $(\vec u,\vec v)$ est une base du plan vectoriel $\mathrm{Vect}(\vec u,\vec v)$.
\end{propriete}

\begin{exemple}
$\vec u=(1,0,1)$, $\vec v=(0,1,1)$, $\vec w=(1,1,2)$ dans $\R^3$. On cherche $a,b$ avec
$\vec w=a\vec u+b\vec v=(a,b,a+b)$. Il faut $a=1$, $b=1$, et $a+b=2$ : c'est vérifié. Donc
$\vec w=\vec u+\vec v$ : les trois vecteurs sont \textbf{coplanaires} (famille liée).
\end{exemple}

\section{Bases, repères et coordonnées}

\begin{definition}[Base]
Une famille $(\vec{e_1},\dots,\vec{e_n})$ est une \textbf{base} de $E$ si elle est à la fois
\textbf{libre} et \textbf{génératrice}. Tout vecteur $\vec u$ de $E$ se décompose alors de façon
\textbf{unique} :
\[
\vec u=x_1\vec{e_1}+\dots+x_n\vec{e_n},
\]
les réels $x_1,\dots,x_n$ étant les \textbf{coordonnées} de $\vec u$ dans cette base.
\end{definition}

\begin{theoreme}[Dimension]
Dans un espace vectoriel $E$ admettant une base finie, toutes les bases ont le \textbf{même nombre}
de vecteurs : ce nombre est la \textbf{dimension} de $E$, notée $\dim E$.
\[
\dim\R^2=2,\qquad \dim\R^3=3,\qquad \dim\R^n=n.
\]
\end{theoreme}

\begin{propriete}[Reconnaître une base par la dimension]
Si $\dim E=n$, alors :
\begin{itemize}
\item toute famille \textbf{libre} de $n$ vecteurs de $E$ est une base ;
\item toute famille \textbf{génératrice} de $n$ vecteurs de $E$ est une base.
\end{itemize}
Il suffit donc de vérifier \emph{une} des deux propriétés quand on a déjà le bon nombre de vecteurs.
\end{propriete}

\begin{definition}[Repère]
Un \textbf{repère} du plan est un triplet $(O,\vec i,\vec j)$ où $O$ est un point (l'origine) et
$(\vec i,\vec j)$ une base. Tout point $M$ a alors des \textbf{coordonnées} $(x,y)$ définies par
$\vec{OM}=x\vec i+y\vec j$. De même, un repère de l'espace est $(O,\vec i,\vec j,\vec k)$ et
$\vec{OM}=x\vec i+y\vec j+z\vec k$ donne les coordonnées $(x,y,z)$ de $M$.
\end{definition}

\begin{exemple}
La \textbf{base canonique} de $\R^2$ est $(\vec i,\vec j)$ avec $\vec i=(1,0)$, $\vec j=(0,1)$ ;
celle de $\R^3$ est $(\vec i,\vec j,\vec k)$ avec $\vec i=(1,0,0)$, $\vec j=(0,1,0)$, $\vec k=(0,0,1)$.
\end{exemple}

\begin{propriete}[Calculs en coordonnées]
Dans un repère, si $A(x_A,y_A)$ et $B(x_B,y_B)$ alors
\[
\vec{AB}=(x_B-x_A,\ y_B-y_A),\qquad
I=\text{milieu de }[AB]\ \Rightarrow\ I\Big(\tfrac{x_A+x_B}{2},\ \tfrac{y_A+y_B}{2}\Big).
\]
Dans un repère \textbf{orthonormé}, $\|\vec{AB}\|=\sqrt{(x_B-x_A)^2+(y_B-y_A)^2}$
(idem dans l'espace avec la troisième coordonnée).
\end{propriete}

Pour une famille de $\R^2$ : \textbf{libre} $\Leftrightarrow$ vecteurs non colinéaires ;
\textbf{base} $\Leftrightarrow$ deux vecteurs non colinéaires. Pour $\R^3$ : \textbf{base}
$\Leftrightarrow$ trois vecteurs non coplanaires. Coordonnées dans une base = \textbf{uniques}.
\end{aretenir}

\section{Déterminant de deux vecteurs du plan}

\begin{definition}[Déterminant]
Soient $\vec u=(x,y)$ et $\vec v=(x',y')$ deux vecteurs de $\R^2$ exprimés dans une base. Le
\textbf{déterminant} de $(\vec u,\vec v)$ est le réel
\[
\det(\vec u,\vec v)=\begin{vmatrix}x&x'\\ y&y'\end{vmatrix}=x\,y'-y\,x'.
\]
\end{definition}

\begin{theoreme}[Colinéarité et déterminant]
Deux vecteurs $\vec u,\vec v$ du plan sont \textbf{colinéaires} si et seulement si
\[
\det(\vec u,\vec v)=xy'-yx'=0.
\]
Par conséquent, $(\vec u,\vec v)$ est une \textbf{base} de $\R^2$ si et seulement si
$\det(\vec u,\vec v)\neq0$.
\end{theoreme}

\begin{remarque}
En repère \textbf{orthonormé}, $|\det(\vec u,\vec v)|$ est l'\textbf{aire} du parallélogramme
construit sur $\vec u$ et $\vec v$. L'aire du triangle $ABC$ vaut alors
$\tfrac12\,\big|\det(\vec{AB},\vec{AC})\big|$. Le déterminant nul $\Leftrightarrow$ aire nulle
$\Leftrightarrow$ points alignés.
\end{remarque}

\begin{exemple}
$\vec u=(2,3)$, $\vec v=(4,6)$ : $\det=2\times6-3\times4=12-12=0$ : colinéaires.\;
$\vec u=(2,3)$, $\vec w=(1,5)$ : $\det=2\times5-3\times1=10-3=7\neq0$ : base de $\R^2$.
\end{exemple}

\begin{propriete}[Propriétés du déterminant]
Le déterminant est \textbf{antisymétrique} et \textbf{linéaire} en chaque vecteur :
\[
\det(\vec v,\vec u)=-\det(\vec u,\vec v),\qquad
\det(\lambda\vec u,\vec v)=\lambda\det(\vec u,\vec v),\qquad
\det(\vec u+\vec u',\vec v)=\det(\vec u,\vec v)+\det(\vec u',\vec v).
\]
En particulier $\det(\vec u,\vec u)=0$.
\end{propriete}

\section{Applications géométriques}

\begin{propriete}[Alignement et parallélisme]
Soient $A,B,C,D$ des points du plan.
\begin{itemize}
\item $A,B,C$ sont \textbf{alignés} $\Leftrightarrow$ $\vec{AB}$ et $\vec{AC}$ sont colinéaires
$\Leftrightarrow$ $\det(\vec{AB},\vec{AC})=0$.
\item Les droites $(AB)$ et $(CD)$ sont \textbf{parallèles} $\Leftrightarrow$ $\vec{AB}$ et $\vec{CD}$
sont colinéaires $\Leftrightarrow$ $\det(\vec{AB},\vec{CD})=0$.
\item $ABDC$ est un \textbf{parallélogramme} $\Leftrightarrow$ $\vec{AB}=\vec{CD}$.
\end{itemize}
\end{propriete}

\begin{exemple}
$A(1,2)$, $B(3,5)$, $C(5,8)$. $\vec{AB}=(2,3)$, $\vec{AC}=(4,6)$ ;
$\det(\vec{AB},\vec{AC})=2\times6-3\times4=0$ : les points $A,B,C$ sont \textbf{alignés}.
\end{exemple}

\begin{propriete}[Équation cartésienne d'une droite]
Dans un repère, une droite admet une équation $ax+by+c=0$ avec $(a,b)\neq(0,0)$. Un
\textbf{vecteur directeur} en est $\vec u=(-b,a)$. Deux droites $ax+by+c=0$ et $a'x+b'y+c'=0$
sont parallèles $\Leftrightarrow$ $ab'-a'b=0$ (déterminant des vecteurs directeurs nul).
\end{propriete}

\begin{exemple}
Droite $D:2x-3y+1=0$, vecteur directeur $\vec u=(3,2)$. Droite $D':4x-6y-5=0$, vecteur directeur
$\vec u'=(6,4)=2\vec u$ : $D\parallel D'$ (et $D\neq D'$ car $c$ diffère).
\end{exemple}

\section{Systèmes d'équations linéaires}

\begin{definition}[Système linéaire $2\times2$]
Un système linéaire de deux équations à deux inconnues s'écrit
\[
\left\{\begin{aligned} ax+by&=e\\ cx+dy&=f \end{aligned}\right.
\qquad\text{de déterminant}\qquad \Delta=\begin{vmatrix}a&b\\ c&d\end{vmatrix}=ad-bc.
\]
\end{definition}

\begin{theoreme}[Formules de Cramer]
Si $\Delta=ad-bc\neq0$, le système admet une \textbf{unique} solution donnée par
\[
x=\frac{\Delta_x}{\Delta}=\frac{1}{\Delta}\begin{vmatrix}e&b\\ f&d\end{vmatrix}=\frac{ed-bf}{ad-bc},
\qquad
y=\frac{\Delta_y}{\Delta}=\frac{1}{\Delta}\begin{vmatrix}a&e\\ c&f\end{vmatrix}=\frac{af-ec}{ad-bc}.
\]
\end{theoreme}

\begin{remarque}[Interprétation et cas $\Delta=0$]
Chaque équation représente une droite. Si $\Delta\neq0$, les droites se coupent en un point :
\textbf{une} solution. Si $\Delta=0$, les droites sont parallèles : soit \textbf{confondues}
(une infinité de solutions), soit \textbf{strictement parallèles} (aucune solution). Le lien avec
le calcul vectoriel est direct : $\Delta$ est le déterminant des vecteurs colonnes $(a,c)$ et
$(b,d)$.
\end{remarque}

\begin{exemple}
$\left\{\begin{aligned}2x+3y&=8\\ x-y&=-1\end{aligned}\right.$ : $\Delta=2(-1)-3(1)=-5\neq0$.
$x=\dfrac{1}{-5}\begin{vmatrix}8&3\\-1&-1\end{vmatrix}=\dfrac{-8+3}{-5}=1$,\;
$y=\dfrac{1}{-5}\begin{vmatrix}2&8\\1&-1\end{vmatrix}=\dfrac{-2-8}{-5}=2$. Solution $(1,2)$.
\end{exemple}

\begin{remarque}[Systèmes $3\times3$]
Un système de trois équations à trois inconnues se résout par \textbf{substitution} ou
\textbf{combinaisons} (méthode du pivot de Gauss) : on élimine successivement les inconnues pour
se ramener à une forme triangulaire. Trois plans peuvent se couper en un point (solution unique),
selon une droite ou un plan (infinité), ou n'avoir aucun point commun.
\end{remarque}

\begin{aretenir}
\textbf{Système $2\times2$.} Calculer $\Delta=ad-bc$. Si $\Delta\neq0$ : solution unique par
Cramer. Si $\Delta=0$ : comparer les équations (proportionnelles $\to$ infinité ; sinon $\to$ pas
de solution). Toujours \textbf{vérifier} la solution dans les deux équations de départ.
\end{aretenir}

\section{Applications linéaires}

\begin{definition}[Application linéaire]
Soient $E$ et $F$ deux espaces vectoriels réels. Une application $f:E\to F$ est \textbf{linéaire}
si, pour tous $\vec u,\vec v\in E$ et tout $\lambda\in\R$ :
\[
f(\vec u+\vec v)=f(\vec u)+f(\vec v)\qquad\text{et}\qquad f(\lambda\vec u)=\lambda f(\vec u).
\]
De façon équivalente : $f(\lambda\vec u+\mu\vec v)=\lambda f(\vec u)+\mu f(\vec v)$ pour tous
$\lambda,\mu\in\R$. On a alors nécessairement $f(\vec 0)=\vec 0$.
\end{definition}

\begin{exemple}
$f:\R^2\to\R^2$, $f(x,y)=(2x-y,\,x+3y)$ est linéaire. L'application $g(x,y)=(x+1,y)$ ne l'est
pas, car $g(0,0)=(1,0)\neq(0,0)$.
\end{exemple}

\begin{definition}[Noyau et image]
Pour $f:E\to F$ linéaire :
\[
\ker f=\{\vec u\in E\mid f(\vec u)=\vec 0\}\ \subset E\quad\text{(le \textbf{noyau})},\qquad
\mathrm{Im}\,f=\{f(\vec u)\mid \vec u\in E\}\ \subset F\quad\text{(l'\textbf{image})}.
\]
\end{definition}

\begin{theoreme}[Propriétés du noyau et de l'image]
Si $f:E\to F$ est linéaire, $\ker f$ est un sous-espace vectoriel de $E$ et $\mathrm{Im}\,f$ un
sous-espace vectoriel de $F$. De plus :
\[
f\ \text{injective}\ \Longleftrightarrow\ \ker f=\{\vec 0\};\qquad
f\ \text{surjective}\ \Longleftrightarrow\ \mathrm{Im}\,f=F.
\]
\end{theoreme}

\begin{propriete}[Théorème du rang]
Si $E$ est de dimension finie et $f:E\to F$ linéaire, en posant
$\mathrm{rg}(f)=\dim(\mathrm{Im}\,f)$ (le \textbf{rang} de $f$) :
\[
\dim(\ker f)+\mathrm{rg}(f)=\dim E.
\]
\end{propriete}

\begin{remarque}
L'image $\mathrm{Im}\,f$ est engendrée par les images des vecteurs d'une base de $E$ : si
$(\vec{e_1},\dots,\vec{e_n})$ est une base de $E$, alors
$\mathrm{Im}\,f=\mathrm{Vect}\big(f(\vec{e_1}),\dots,f(\vec{e_n})\big)$.
\end{remarque}

\section{L'essentiel à retenir}

\begin{synthese}
\textbf{Vecteurs.} $\vec{AB}+\vec{BC}=\vec{AC}$ (Chasles) ;\ $\vec{AB}=(x_B-x_A,\,y_B-y_A)$ ;\
milieu $I\big(\tfrac{x_A+x_B}2,\tfrac{y_A+y_B}2\big)$ ;\ norme (orthonormé)
$\sqrt{(x_B-x_A)^2+(y_B-y_A)^2}$.\\[4pt]
\textbf{Espace vectoriel.} $\R^n$ ; s.e.v. $F$ : $\vec0\in F$ et $\lambda\vec u+\mu\vec v\in F$.
Les s.e.v. de $\R^3$ : $\{\vec0\}$, droites, plans, $\R^3$.\\[4pt]
\textbf{Combinaison linéaire.} $\sum\lambda_i\vec{u_i}$ ; famille \textbf{génératrice} si tout
vecteur s'écrit ainsi ($\mathrm{Vect}$).\\[4pt]
\textbf{Libre / liée.} Libre $\Leftrightarrow$ ($\sum\lambda_i\vec{u_i}=\vec0\Rightarrow$ tous
$\lambda_i=0$). Deux vecteurs : libre $\Leftrightarrow$ non colinéaires. Trois vecteurs de $\R^3$ :
libre $\Leftrightarrow$ non coplanaires.\\[4pt]
\textbf{Base \& dimension.} Base = libre + génératrice ; coordonnées \textbf{uniques}.
$\dim\R^2=2$, $\dim\R^3=3$. Avec le bon nombre de vecteurs, libre $\Rightarrow$ base.\\[4pt]
\textbf{Déterminant (plan).} $\det\big((x,y),(x',y')\big)=xy'-yx'$. Colinéaires $\Leftrightarrow$
$\det=0$ ; base $\Leftrightarrow$ $\det\neq0$ ; $|\det|=$ aire du parallélogramme.\\[4pt]
\textbf{Applications géom.} $A,B,C$ alignés $\Leftrightarrow$ $\det(\vec{AB},\vec{AC})=0$ ;
$(AB)\parallel(CD)\Leftrightarrow\det(\vec{AB},\vec{CD})=0$ ;\ $ABDC$ parallélogramme
$\Leftrightarrow\vec{AB}=\vec{CD}$. Droite $ax+by+c=0$, directeur $(-b,a)$.\\[4pt]
\textbf{Système $2\times2$.} $\Delta=ad-bc$. Cramer : $x=\Delta_x/\Delta$, $y=\Delta_y/\Delta$.
$\Delta=0\Rightarrow$ aucune ou une infinité de solutions.\\[4pt]
\textbf{Appli. linéaire.} $f(\lambda\vec u+\mu\vec v)=\lambda f(\vec u)+\mu f(\vec v)$ ;\
$\ker f$, $\mathrm{Im}\,f$ s.e.v. ; injective $\Leftrightarrow\ker f=\{\vec0\}$ ; théorème du rang
$\dim\ker f+\mathrm{rg}(f)=\dim E$.\\[4pt]
\textbf{Pièges.} oublier $\vec0\in F$ ; confondre colinéaire (2 vect.) et coplanaire (3 vect.) ;
oublier de vérifier la solution d'un système ; conclure « linéaire » sans tester $f(\vec0)=\vec0$.
\end{synthese}

\section{Méthodes \& savoir-faire}

\methode{Calculer avec des vecteurs en coordonnées}
Utiliser $\vec{AB}=(x_B-x_A,\,y_B-y_A)$, l'addition composante par composante et la relation de
Chasles pour simplifier les expressions.
\begin{exemple}
$A(2,-1)$, $B(5,3)$, $C(0,4)$. Alors $\vec{AB}=(3,4)$, $\vec{AC}=(-2,5)$ et
$\vec{AB}+\vec{AC}=(1,9)$. Le milieu de $[BC]$ est $\big(\tfrac{5+0}2,\tfrac{3+4}2\big)=(2{,}5\,;\,3{,}5)$.
\end{exemple}

\methode{Tester la colinéarité de deux vecteurs}
Calculer le déterminant $\det(\vec u,\vec v)=xy'-yx'$ : nul $\Rightarrow$ colinéaires, non nul
$\Rightarrow$ libres (donc base de $\R^2$).
\begin{exemple}
$\vec u=(6,-4)$, $\vec v=(-3,2)$ : $\det=6\times2-(-4)\times(-3)=12-12=0$ : colinéaires
(en effet $\vec v=-\tfrac12\vec u$).
\end{exemple}

\methode{Montrer que trois points sont alignés}
Former deux vecteurs $\vec{AB}$ et $\vec{AC}$ ; les points sont alignés ssi
$\det(\vec{AB},\vec{AC})=0$.
\begin{exemple}
$A(1,1)$, $B(4,3)$, $C(-2,-1{,}\overline{6})$ ? Prenons $A(1,1)$, $B(4,3)$, $C(7,5)$ :
$\vec{AB}=(3,2)$, $\vec{AC}=(6,4)$, $\det=3\times4-2\times6=0$ : alignés.
\end{exemple}

\methode{Montrer qu'un ensemble est un sous-espace vectoriel}
Vérifier les trois points : $\vec 0\in F$, puis pour $\vec u,\vec v\in F$ et $\lambda,\mu\in\R$,
montrer $\lambda\vec u+\mu\vec v\in F$ (caractérisation pratique).
\begin{exemple}
$F=\{(x,y,z)\in\R^3\mid x+y-z=0\}$. \;$\vec 0=(0,0,0)$ vérifie $0+0-0=0$ donc $\vec 0\in F$.
Soient $\vec u=(x,y,z)$, $\vec v=(x',y',z')$ dans $F$ et $\lambda,\mu\in\R$. Posons
$\vec w=\lambda\vec u+\mu\vec v=(\lambda x+\mu x',\ \lambda y+\mu y',\ \lambda z+\mu z')$.
Alors $(\lambda x+\mu x')+(\lambda y+\mu y')-(\lambda z+\mu z')=\lambda(x+y-z)+\mu(x'+y'-z')=0$.
Donc $\vec w\in F$ : $F$ est un s.e.v. de $\R^3$ (un \textbf{plan vectoriel}).
\end{exemple}

\methode{Étudier la liberté d'une famille}
Poser $\lambda_1\vec{u_1}+\dots+\lambda_p\vec{u_p}=\vec 0$, traduire en système et résoudre :
famille libre si la seule solution est nulle, liée sinon.
\begin{exemple}
$\vec u=(1,1,0)$, $\vec v=(0,1,1)$ dans $\R^3$. $\lambda\vec u+\mu\vec v=(\lambda,\lambda+\mu,\mu)=(0,0,0)$
donne $\lambda=0$, $\mu=0$ : la famille est \textbf{libre}.
\end{exemple}

\methode{Décomposer un vecteur dans une base}
Écrire $\vec w=a\vec u+b\vec v$, identifier composante par composante, résoudre le système
linéaire en $(a,b)$ (par Cramer ou substitution).
\begin{exemple}
Décomposer $\vec w=(5,5)$ dans la base $\big((1,2),(3,1)\big)$ : $a+3b=5$, $2a+b=5$. De la
seconde $b=5-2a$ ; report $a+15-6a=5\Rightarrow a=2$, $b=1$. Donc $\vec w=2(1,2)+(3,1)$.
\end{exemple}

\methode{Résoudre un système linéaire $2\times2$ par Cramer}
Calculer $\Delta=ad-bc$. Si $\Delta\neq0$, appliquer $x=\Delta_x/\Delta$, $y=\Delta_y/\Delta$,
puis \textbf{vérifier}.
\begin{exemple}
$\left\{\begin{aligned}3x+2y&=7\\ x-4y&=-1\end{aligned}\right.$ : $\Delta=3(-4)-2(1)=-14$.
$x=\dfrac{1}{-14}\begin{vmatrix}7&2\\-1&-4\end{vmatrix}=\dfrac{-28+2}{-14}=\dfrac{-26}{-14}=\dfrac{13}{7}$,\;
$y=\dfrac{1}{-14}\begin{vmatrix}3&7\\1&-1\end{vmatrix}=\dfrac{-3-7}{-14}=\dfrac{5}{7}$.
\end{exemple}

\methode{Montrer qu'une application est linéaire et trouver son noyau}
Vérifier $f(\lambda\vec u+\mu\vec v)=\lambda f(\vec u)+\mu f(\vec v)$ ; un contre-exemple (souvent
$f(\vec 0)\neq\vec 0$) suffit pour la non-linéarité. Pour $\ker f$, résoudre $f(\vec u)=\vec0$.
\begin{exemple}
$f(x,y)=(x-y,\,2x-2y)$. Linéarité immédiate (composantes affines sans terme constant).
$f(x,y)=(0,0)\Leftrightarrow x-y=0$ : $\ker f=\mathrm{Vect}\big((1,1)\big)$, $\dim\ker f=1$.
\end{exemple}

\methode{Déterminer l'image et le rang}
Calculer $f$ sur une base de départ : $\mathrm{Im}\,f$ est engendrée par ces images ; en extraire
une famille libre pour la base et le rang.
\begin{exemple}
Avec $f(x,y)=(x-y,2x-2y)$ : $f(1,0)=(1,2)$, $f(0,1)=(-1,-2)=-(1,2)$. Donc
$\mathrm{Im}\,f=\mathrm{Vect}\big((1,2)\big)$, $\mathrm{rg}(f)=1$. Vérif. du rang : $1+1=2=\dim\R^2$. \checkmark
\end{exemple}

\item Montrer que $(\vec u,\vec v)$ est une base de $\R^2$ (on calculera un déterminant).
\item Soit $\vec w=(5,5)$. Déterminer les coordonnées $(a,b)$ de $\vec w$ dans la base
$(\vec u,\vec v)$ à l'aide des formules de Cramer.
\item Vérifier le résultat en calculant $a\vec u+b\vec v$.
\item Soit $g:\R^2\to\R^2$ définie par $g(x,y)=(x+3y,\,2x-y)$. Montrer que $g$ est linéaire et
exprimer $g(\vec u)$ et $g(\vec v)$.
\item En déduire que $g$ envoie la base $(\vec i,\vec j)$ sur $(\vec u,\vec v)$.
\end{enumerate}

\sujetbac[application linéaire, noyau, image et théorème du rang]
On considère l'application $f:\R^3\to\R^3$ définie par
\[
f(x,y,z)=(x-y,\ y-z,\ z-x).
\]
\begin{enumerate}[label=\arabic*)]
\item Montrer que $f$ est une application linéaire.
\item Déterminer le noyau $\ker f$ ; en donner une base et la dimension.
\item En déduire le rang de $f$ à l'aide du théorème du rang. $f$ est-elle bijective ?
\item Calculer $f(1,0,0)$, $f(0,1,0)$, $f(0,0,1)$ et en déduire une base de $\mathrm{Im}\,f$.
\item Vérifier que $\mathrm{Im}\,f=\{(a,b,c)\in\R^3\mid a+b+c=0\}$.
\end{enumerate}

\section{Corrigés}

\corrige{de l'exercice 1}
$\vec{AB}=(3-(-1),-1-2)=(4,-3)$ ; $\vec{BC}=(2-3,4-(-1))=(-1,5)$ ;
$\vec{AC}=(2-(-1),4-2)=(3,2)$. Vérification : $\vec{AB}+\vec{BC}=(4-1,-3+5)=(3,2)=\vec{AC}$. \checkmark

\corrige{de l'exercice 2}
$ABCD$ parallélogramme $\Leftrightarrow\vec{AB}=\vec{DC}$, soit $\vec{AD}=\vec{BC}$.
$\vec{BC}=(1,3)$, donc $D=A+\vec{BC}=(0+1,1+3)=(1,4)$. (On vérifie $\vec{AB}=(4,1)=\vec{DC}=(5-1,5-4)=(4,1)$.)

\corrige{de l'exercice 3}
$\vec v=(-6,3)=-3\,(2,-1)=-3\vec u$ : $\vec u$ et $\vec v$ sont colinéaires, donc la famille est
\textbf{liée}.

\corrige{de l'exercice 4}
$\vec{AB}=(3,1)$, $\vec{AC}=(6,2)$. $\det(\vec{AB},\vec{AC})=3\times2-1\times6=0$ : les points
$A,B,C$ sont \textbf{alignés}.

\corrige{de l'exercice 5}
$\vec{AB}=(2,2)$, $\vec{CD}=(4,4)$. $\det(\vec{AB},\vec{CD})=2\times4-2\times4=0$ : $\vec{AB}$ et
$\vec{CD}$ sont colinéaires, donc $(AB)\parallel(CD)$.

\corrige{de l'exercice 6}
$A$ : droite passant par l'origine ($(0,0)\in A$, stable) $\Rightarrow$ \textbf{s.e.v.}\;
$B$ : $(0,0)\notin B$ car $0\neq0+1$ $\Rightarrow$ \textbf{non}.\;
$C$ : $x^2=y^2\Leftrightarrow y=\pm x$, réunion de deux droites ; $(1,1)\in C$ et $(1,-1)\in C$
mais leur somme $(2,0)\notin C$ (car $4\neq0$) $\Rightarrow$ \textbf{non} (pas stable par addition).

\corrige{de l'exercice 7}
$\vec 0=(0,0,0)$ : $2\cdot0-0+0=0$ donc $\vec 0\in F$. Soient $\vec u=(x,y,z)$, $\vec v=(x',y',z')$
dans $F$ et $\lambda,\mu\in\R$. Pour $\vec w=\lambda\vec u+\mu\vec v$ :
$2(\lambda x+\mu x')-(\lambda y+\mu y')+(\lambda z+\mu z')=\lambda(2x-y+z)+\mu(2x'-y'+z')=0$.
Donc $\vec w\in F$ : $F$ est un sous-espace vectoriel de $\R^3$ (un plan vectoriel).

\corrige{de l'exercice 8}
$(1,2)$ et $(3,1)$ : $\det=1\times1-2\times3=-5\neq0$, donc famille libre de $2$ vecteurs dans
$\R^2$ : c'est une \textbf{base}. On cherche $a,b$ avec $a(1,2)+b(3,1)=(5,5)$, soit $a+3b=5$ et
$2a+b=5$. De la seconde $b=5-2a$ ; report : $a+15-6a=5\Rightarrow-5a=-10\Rightarrow a=2$, $b=1$.
Coordonnées : $\vec w=2(1,2)+1(3,1)$, soit $(2,1)$ dans cette base.

\corrige{de l'exercice 9}
$F=\{(2y,y,0)\mid y\in\R\}=y\,(2,1,0)$. Le vecteur $(2,1,0)$ engendre $F$ et est non nul (donc
libre) : c'est une \textbf{base} de $F$, et $\dim F=1$ ($F$ est une droite vectorielle).

\corrige{de l'exercice 10}
Trois vecteurs dans $\R^3$ ($\dim=3$) : il suffit de montrer qu'ils sont libres. Posons
$\lambda(1,1,0)+\mu(0,1,1)+\nu(1,0,1)=(0,0,0)$, soit
$\lambda+\nu=0$, $\lambda+\mu=0$, $\mu+\nu=0$. De la première $\nu=-\lambda$ ; de la deuxième
$\mu=-\lambda$ ; report dans la troisième : $-\lambda-\lambda=0\Rightarrow\lambda=0$, puis
$\mu=\nu=0$. La famille est libre, donc c'est une \textbf{base} de $\R^3$.

\corrige{de l'exercice 11}
On cherche $a,b$ avec $\vec w=a\vec u+b\vec v$, soit $(0,1,-2)=a(1,0,1)+b(2,1,0)=(a+2b,\,b,\,a)$.
On a $b=1$, $a=-2$, et il faut $a+2b=0$ : $-2+2=0$ \checkmark. Donc $\vec w=-2\vec u+\vec v$ : les
trois vecteurs sont \textbf{coplanaires} (famille liée).

\corrige{de l'exercice 12}
$\det\big((3,-2),(1,4)\big)=3\times4-(-2)\times1=12+2=14\neq0$ : libres (base de $\R^2$).\;
$\det\big((2,6),(1,3)\big)=2\times3-6\times1=0$ : colinéaires.

\corrige{de l'exercice 13}
$\Delta=2(-1)-3(1)=-5\neq0$. $x=\dfrac{1}{-5}\begin{vmatrix}8&3\\-1&-1\end{vmatrix}=\dfrac{-8+3}{-5}=1$,\;
$y=\dfrac{1}{-5}\begin{vmatrix}2&8\\1&-1\end{vmatrix}=\dfrac{-2-8}{-5}=2$. Solution $(1,2)$
(on vérifie : $2(1)+3(2)=8$ et $1-2=-1$). \checkmark

\corrige{de l'exercice 14}
$\Delta=\begin{vmatrix}1&2\\2&m\end{vmatrix}=m-4$.\;
\textbf{Si $m\neq4$} : $\Delta\neq0$, solution unique (Cramer).\;
\textbf{Si $m=4$} : le système devient $x+2y=3$ et $2x+4y=6$, la seconde équation est le double
de la première : \textbf{une infinité} de solutions (droite $x+2y=3$).

\corrige{de l'exercice 15}
$f(x,y)=(3x,x-y)$ : pour $\vec u=(x,y)$, $\vec v=(x',y')$, $\lambda,\mu\in\R$,
$f(\lambda\vec u+\mu\vec v)=\big(3(\lambda x+\mu x'),(\lambda x+\mu x')-(\lambda y+\mu y')\big)
=\lambda(3x,x-y)+\mu(3x',x'-y')=\lambda f(\vec u)+\mu f(\vec v)$ : \textbf{linéaire}.\;
$g(x,y)=(xy,x)$ : $g(1,1)=(1,1)$ donc $2g(1,1)=(2,2)$, tandis que $g(2,2)=(4,2)\neq(2,2)$ :
\textbf{non linéaire}.

\corrige{de l'exercice 16}
\textbf{Linéarité} : $f(\lambda\vec u+\mu\vec v)=\big((\lambda x+\mu x')+(\lambda y+\mu y')-(\lambda z+\mu z'),\,2(\lambda x+\mu x')-(\lambda z+\mu z')\big)
=\lambda f(\vec u)+\mu f(\vec v)$ : $f$ est linéaire.\;
\textbf{Noyau} : $f(x,y,z)=(0,0)\Leftrightarrow x+y-z=0$ et $2x-z=0$, soit $z=2x$ et $y=z-x=x$.
Donc $\ker f=\{(x,x,2x)\mid x\in\R\}=\mathrm{Vect}\big((1,1,2)\big)$, droite vectorielle de
dimension $1$.

\corrige{de l'exercice 17}
$f(x,y)=(x+2y,2x+4y)$.\;
\textbf{Linéarité} : composantes linéaires en $(x,y)$, donc $f$ est linéaire.\;
\textbf{Noyau} : $x+2y=0$ et $2x+4y=0$ (double de la première) ; il reste $x=-2y$, d'où
$\ker f=\mathrm{Vect}\big((-2,1)\big)$, base $\{(-2,1)\}$, $\dim\ker f=1$. Donc $f$ \textbf{n'est pas
injective}.\;
\textbf{Image} : $f(1,0)=(1,2)$, $f(0,1)=(2,4)=2(1,2)$, donc
$\mathrm{Im}\,f=\mathrm{Vect}\big((1,2)\big)$, $\mathrm{rg}(f)=1$. Vérif. : $1+1=2=\dim\R^2$ \checkmark.
Comme $\mathrm{Im}\,f\neq\R^2$, $f$ \textbf{n'est pas surjective}.

\corrige{du sujet 1}
1) $\vec{AB}=(2-(-1),3-2)=(3,1)$, $\vec{AC}=(5-(-1),4-2)=(6,2)$.\\
2) $\det(\vec{AB},\vec{AC})=3\times2-1\times6=0$ : les points $A,B,C$ sont \textbf{alignés}.\\
3) $\vec{AD}=(1-(-1),-1-2)=(2,-3)$. $\det(\vec{AB},\vec{AD})=3\times(-3)-1\times2=-11$. Aire du
triangle $ABD=\tfrac12|{-11}|=\dfrac{11}{2}=5{,}5$ unités d'aire.\\
4) $ABED$ parallélogramme $\Leftrightarrow\vec{AB}=\vec{DE}$, donc $E=D+\vec{AB}=(1+3,-1+1)=(4,0)$.\\
5) $\vec{CD}=(1-5,-1-4)=(-4,-5)$. $\det(\vec{AB},\vec{CD})=3\times(-5)-1\times(-4)=-15+4=-11\neq0$ :
$(AB)$ et $(CD)$ \textbf{ne sont pas parallèles}.

\corrige{du sujet 2}
1) $\det(\vec u,\vec v)=\begin{vmatrix}1&3\\2&-1\end{vmatrix}=1\times(-1)-2\times3=-7\neq0$ :
$(\vec u,\vec v)$ est une \textbf{base} de $\R^2$.\\
2) On résout $a\vec u+b\vec v=(5,5)$, soit $a+3b=5$ et $2a-b=5$, de déterminant $\Delta=-7$.
$a=\dfrac{1}{-7}\begin{vmatrix}5&3\\5&-1\end{vmatrix}=\dfrac{-5-15}{-7}=\dfrac{-20}{-7}=\dfrac{20}{7}$,\;
$b=\dfrac{1}{-7}\begin{vmatrix}1&5\\2&5\end{vmatrix}=\dfrac{5-10}{-7}=\dfrac{-5}{-7}=\dfrac{5}{7}$.\\
3) $a\vec u+b\vec v=\tfrac{20}{7}(1,2)+\tfrac57(3,-1)=\big(\tfrac{20+15}{7},\tfrac{40-5}{7}\big)
=\big(\tfrac{35}{7},\tfrac{35}{7}\big)=(5,5)$. \checkmark\\
4) $g$ a des composantes linéaires en $(x,y)$ : $g(\lambda\vec a+\mu\vec b)=\lambda g(\vec a)+\mu g(\vec b)$,
donc $g$ est \textbf{linéaire}. $g(\vec u)=g(1,2)=(1+6,2-2)=(7,0)$ ;
$g(\vec v)=g(3,-1)=(3-3,6+1)=(0,7)$.\\
5) $g(\vec i)=g(1,0)=(1,2)=\vec u$ et $g(\vec j)=g(0,1)=(3,-1)=\vec v$. Donc $g$ envoie la base
canonique $(\vec i,\vec j)$ sur la base $(\vec u,\vec v)$.

\corrige{du sujet 3}
1) $f(x,y,z)=(x-y,y-z,z-x)$ : chaque composante est linéaire en $(x,y,z)$, donc pour tous
$\vec U,\vec V$ et $\lambda,\mu$, $f(\lambda\vec U+\mu\vec V)=\lambda f(\vec U)+\mu f(\vec V)$ :
$f$ est \textbf{linéaire}.\\
2) $f(x,y,z)=(0,0,0)\Leftrightarrow x-y=0,\ y-z=0,\ z-x=0\Leftrightarrow x=y=z$. Donc
$\ker f=\{(x,x,x)\mid x\in\R\}=\mathrm{Vect}\big((1,1,1)\big)$ : base $\{(1,1,1)\}$, $\dim\ker f=1$.\\
3) Théorème du rang : $\mathrm{rg}(f)=\dim\R^3-\dim\ker f=3-1=2$. Comme $\ker f\neq\{\vec0\}$,
$f$ n'est pas injective, donc \textbf{pas bijective}.\\
4) $f(1,0,0)=(1,0,-1)$, $f(0,1,0)=(-1,1,0)$, $f(0,0,1)=(0,-1,1)$. Ces vecteurs engendrent
$\mathrm{Im}\,f$ ; les deux premiers sont non colinéaires (donc libres) et le rang vaut $2$ : une
base de $\mathrm{Im}\,f$ est $\big((1,0,-1),(-1,1,0)\big)$.\\
5) Pour tout $(a,b,c)=f(x,y,z)=(x-y,y-z,z-x)$ : $a+b+c=(x-y)+(y-z)+(z-x)=0$. Donc
$\mathrm{Im}\,f\subset\{a+b+c=0\}$. Ce dernier est un plan (dimension $2$) contenant
$\mathrm{Im}\,f$ qui est aussi de dimension $2$ : il y a égalité,
$\mathrm{Im}\,f=\{(a,b,c)\mid a+b+c=0\}$.
