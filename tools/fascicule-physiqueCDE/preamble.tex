% ════════════════════════════════════════════════════════════════
%  PLACEHOLDER D'IMAGE EXTERNE  — \imgph{cle}{description}
%  Le modèle NE DESSINE PAS les schémas trop complexes (montages réels,
%  optique fine) : il pose ce placeholder en décrivant précisément l'image
%  attendue. Un script (fetch_images.mjs) collecte tous les \imgph, télécharge
%  une image libre correspondante dans images/<cle>.* puis remplace le
%  placeholder par \imgreal{<cle>}{<légende>}.
%  - <cle>  : identifiant unique, sert de nom de fichier (ex. ch09-fentes-young)
%  - <description> : phrase précise en français (sujet, éléments, vue) pour la recherche d'image.
% ════════════════════════════════════════════════════════════════
\newtcolorbox{imgphbox}{enhanced,colback=onbsoft!60,colframe=onbo2,boxrule=0pt,
  frame hidden,borderline={1pt}{0pt}{onbo2,dashed},
  arc=4pt,left=10pt,right=10pt,top=8pt,bottom=8pt,halign=center,breakable}
\newcommand{\imgph}[2]{%
  \par\smallskip
  \begin{imgphbox}
  {\sffamily\bfseries\color{onbo}[\,IMAGE À INSÉRER\,]}\;{\ttfamily\small\color{onbmuted}#1}\par\smallskip
  {\itshape\color{onbink}#2}
  \end{imgphbox}\par\smallskip}
