\section{Chapitre 1 — Généralités sur la chimie organique}

\corrige{de l'exercice 6 (★★★)}
\begin{enumerate}[label=\arabic*)]
\item Masse de carbone : $m_C = m_{\ce{CO2}}\times\dfrac{12}{44} = 0{,}44\times\dfrac{12}{44} = \SI{0.12}{\gram}$.
Masse d'hydrogène : $m_H = m_{\ce{H2O}}\times\dfrac{2}{18} = 0{,}18\times\dfrac{2}{18} = \SI{0.02}{\gram}$.
Masse d'oxygène : $m_O = m - m_C - m_H = 0{,}30 - 0{,}12 - 0{,}02 = \SI{0.16}{\gram}$.
\item Quantités : $n_C = \dfrac{0{,}12}{12} = \SI{0.01}{\mol}$ ; $n_H = \dfrac{0{,}02}{1} = \SI{0.02}{\mol}$ ;
$n_O = \dfrac{0{,}16}{16} = \SI{0.01}{\mol}$.
On a $n_C : n_H : n_O = 1 : 2 : 1$, d'où le motif $\ce{CH2O}$ de masse molaire \SI{30}{\gram\per\mol}.
Comme $M = \SI{60}{\gram\per\mol} = 2\times30$, la formule brute est $\ce{C2H4O2}$.
\item Le composé est un acide ($\ce{C2H4O2}$ avec un groupe \ce{-COOH}) : c'est
l'\textbf{acide éthanoïque} (acide acétique) $\ce{CH3-COOH}$.
\end{enumerate}

\corrige{du sujet 1}
\begin{enumerate}[label=\arabic*)]
\item Quantité de $A$ vaporisé : $n = \dfrac{V}{V_m} = \dfrac{0{,}24}{24} = \SI{0.01}{\mol}$.
Masse molaire : $M = \dfrac{m}{n} = \dfrac{0{,}58}{0{,}01} = \SI{58}{\gram\per\mol}$.
\item $m_C = 1{,}32\times\dfrac{12}{44} = \SI{0.36}{\gram}$, soit $n_C = \SI{0.03}{\mol}$ ;
$m_H = 0{,}54\times\dfrac{2}{18} = \SI{0.06}{\gram}$, soit $n_H = \SI{0.06}{\mol}$ ;
$m_O = 0{,}58 - 0{,}36 - 0{,}06 = \SI{0.16}{\gram}$, soit $n_O = \SI{0.01}{\mol}$.
Ainsi $n_C : n_H : n_O = 3 : 6 : 1$ : la formule brute est $\ce{C3H6O}$ (vérification :
$M = 3\times12 + 6 + 16 = \SI{58}{\gram\per\mol}$).
\item Isomères de $\ce{C3H6O}$ : le propanal $\ce{CH3-CH2-CHO}$ (aldéhyde), la
propanone $\ce{CH3-CO-CH3}$ (cétone) et le prop-2-én-1-ol $\ce{CH2=CH-CH2-OH}$
(alcool insaturé).
\item $A$ réagit avec la 2,4-DNPH : il possède un groupe carbonyle $\ce{C=O}$ (aldéhyde
ou cétone). Mais il ne réduit pas la liqueur de Fehling : ce n'est pas un aldéhyde.
$A$ est donc une cétone : la \textbf{propanone} $\ce{CH3-CO-CH3}$.
\end{enumerate}

\corrige{du sujet 2}
\begin{enumerate}[label=\arabic*)]
\item Masse molaire : $M = 29\,d = 29\times1{,}93 \approx \SI{56}{\gram\per\mol}$.
\item À pression et température fixées, les volumes sont proportionnels aux quantités
de matière. Le volume de $\ce{CO2}$ formé valant quatre fois celui de $X$, une mole de
$X$ libère quatre moles de $\ce{CO2}$ : la molécule contient donc \textbf{4 atomes de carbone}.
\item Avec 4 carbones : $4\times12 = 48$, donc la masse d'hydrogène par mole est
$56 - 48 = \SI{8}{\gram\per\mol}$, soit 8 atomes d'hydrogène. La formule brute est
$\ce{C4H8}$. Cette formule de type $\ce{C_nH_{2n}}$ correspond à un composé
\textbf{insaturé} : un alcène (but-1-ène, but-2-ène $Z$ ou $E$, méthylpropène) ou un
cycloalcane (cyclobutane).
\end{enumerate}

\section{Chapitre 2 — Les alcanes}

\corrige{de l'exercice ★★★ (mélange méthane–éthane)}
\begin{enumerate}[label=\arabic*)]
\item $\ce{CH4 + 2 O2 -> CO2 + 2 H2O}$ \quad et \quad $\ce{2 C2H6 + 7 O2 -> 4 CO2 + 6 H2O}$.
\item Soit $x$ le volume de méthane et $y$ celui d'éthane. Les volumes étant
proportionnels aux quantités : $x + y = 2{,}0$ et (CO$_2$ formé) $x + 2y = 3{,}0$.
On obtient $y = \SI{1.0}{\litre}$ et $x = \SI{1.0}{\litre}$.
\item Volume de dioxygène : $V_{\ce{O2}} = 2x + 3{,}5\,y = 2{,}0 + 3{,}5 = \SI{5.5}{\litre}$.
Composition molaire : \SI{50}{\percent} de méthane et \SI{50}{\percent} d'éthane.
\end{enumerate}

\corrige{du sujet 1}
\begin{enumerate}[label=\arabic*)]
\item Pour un alcane $\ce{C_nH_{2n+2}}$ : $14n + 2 = 72$ d'où $n = 5$ : formule $\ce{C5H12}$.
\item Isomères : le pentane, le 2-méthylbutane et le 2,2-diméthylpropane.
\item Le 2,2-diméthylpropane $\ce{C(CH3)4}$ ne possède qu'une seule sorte d'atomes
d'hydrogène (tous équivalents) : sa monochloration ne donne qu'un seul produit,
$\ce{(CH3)3C-CH2Cl}$. Équation : $\ce{C5H12 + Cl2 ->[h\nu] C5H11Cl + HCl}$.
\item $n(\ce{C5H12}) = \dfrac{3{,}6}{72} = \SI{0.05}{\mol}$. Combustion :
$\ce{C5H12 + 8 O2 -> 5 CO2 + 6 H2O}$, donc $n(\ce{O2}) = 8\times0{,}05 = \SI{0.40}{\mol}$,
soit $V_{\ce{O2}} = 0{,}40\times24 = \SI{9.6}{\litre}$. L'air contenant \SI{20}{\percent}
de dioxygène : $V_{\text{air}} = \dfrac{9{,}6}{0{,}20} = \SI{48}{\litre}$.
\end{enumerate}

\corrige{du sujet 2}
\begin{enumerate}[label=\arabic*)]
\item $\ce{C3H8 + 5 O2 -> 3 CO2 + 4 H2O}$ \quad et \quad $\ce{2 C4H10 + 13 O2 -> 8 CO2 + 10 H2O}$.
\item Soit $a$ et $b$ les fractions molaires de propane et de butane : $a + b = 1$ et
$5a + 6{,}5\,b = 6{,}3$. On trouve $a = 0{,}13$ et $b = 0{,}87$ : environ
\SI{13}{\percent} de propane et \SI{87}{\percent} de butane.
\item Masse molaire moyenne : $\overline{M} = 0{,}13\times44 + 0{,}87\times58 \approx \SI{56}{\gram\per\mol}$.
Quantité de $\ce{CO2}$ : $3a + 4b = 0{,}40 + 3{,}48 = \SI{3.87}{\mol}$, soit
$m_{\ce{CO2}} = 3{,}87\times44 \approx \SI{170}{\gram}$.
\end{enumerate}

\section{Chapitre 3 — Les alcènes et les alcynes}

\corrige{de l'exercice ★★★ (hydratation de l'éthyne)}
L'hydratation de l'éthyne donne l'éthanal : $\ce{C2H2 + H2O ->[\text{cat.}] CH3-CHO}$.
$B = $ éthanal. Il donne un précipité avec la 2,4-DNPH (présence du groupe carbonyle
$\ce{C=O}$) et réduit la liqueur de Fehling et le réactif de Tollens (c'est un
\emph{aldéhyde}).

\corrige{du sujet 1}
\begin{enumerate}[label=\arabic*)]
\item $M = 29\,d = 29\times1{,}93 \approx \SI{56}{\gram\per\mol}$. L'addition de
dibrome ($1{:}1$) et la décoloration de l'eau de brome montrent une double liaison :
c'est un alcène $\ce{C_nH_{2n}}$, $14n = 56$, $n = 4$ : $\ce{C4H8}$.
\item L'hydratation ne donne qu'un \emph{seul} alcool, tertiaire : l'alcène est
symétrique vis-à-vis de l'addition de Markovnikov, c'est le \textbf{2-méthylpropène}
$\ce{CH2=C(CH3)2}$. $\ce{CH2=C(CH3)2 + H2O -> (CH3)3C-OH}$ (2-méthylpropan-2-ol, tertiaire).
\item Addition de $\ce{HBr}$ (Markovnikov) : $\ce{CH2=C(CH3)2 + HBr -> (CH3)3C-Br}$,
le 2-bromo-2-méthylpropane.
\end{enumerate}

\corrige{du sujet 2}
\begin{enumerate}[label=\arabic*)]
\item $n(\ce{H2}) = \dfrac{1{,}8}{24} = \SI{0.075}{\mol}$ ; $n(X) = \dfrac{2{,}0}{54} = \SI{0.037}{\mol}$.
Le rapport $\dfrac{n(\ce{H2})}{n(X)} = 2$ : chaque molécule fixe 2 molécules de dihydrogène.
\item Deux insaturations : $X$ est un diène ou un alcyne. Avec $M = \SI{54}{\gram\per\mol}$,
formule $\ce{C4H6}$.
\item Par exemple le but-1-yne $\ce{CH#C-CH2-CH3}$ ou le buta-1,3-diène
$\ce{CH2=CH-CH=CH2}$ ; hydrogénation totale : $\ce{C4H6 + 2 H2 -> C4H10}$ (butane).
\end{enumerate}

\section{Chapitre 4 — Les hydrocarbures aromatiques}

\corrige{du sujet 1}
\begin{enumerate}[label=\arabic*)]
\item $M(\ce{C7H8}) = 7\times12 + 8 = \SI{92}{\gram\per\mol}$ ;
$\%\,\ce{C} = \dfrac{84}{92} \approx \SI{91.3}{\percent}$.
\item $\ce{C6H5-CH3 + HNO3 ->[H2SO4] C6H4(CH3)(NO2) + H2O}$ (nitrotoluène, ortho et para).
\item L'acide benzoïque a pour formule $\ce{C6H5-COOH}$, $M = 7\times12 + 6 + 32 = \SI{122}{\gram\per\mol}$.
\item $n(\text{acide}) = \dfrac{12{,}2}{122} = \SI{0.10}{\mol}$. Le rendement étant de
\SI{80}{\percent}, il faut $n(\text{toluène}) = \dfrac{0{,}10}{0{,}80} = \SI{0.125}{\mol}$,
soit $m = 0{,}125\times92 = \SI{11.5}{\gram}$ de toluène.
\end{enumerate}

\corrige{du sujet 2}
\begin{enumerate}[label=\arabic*)]
\item Le TNT est un cycle benzénique portant un groupe méthyle (position 1) et trois
groupes nitro $\ce{-NO2}$ aux positions 2, 4 et 6.
\item $M(\ce{C7H5N3O6}) = 7\times12 + 5 + 3\times14 + 6\times16 = \SI{227}{\gram\per\mol}$.
\item $\ce{C6H5-CH3 + 3 HNO3 ->[H2SO4] C6H2(CH3)(NO2)3 + 3 H2O}$.
\end{enumerate}

\section{Chapitre 5 — Les alcools et les phénols}

\corrige{de l'exercice ★★★ (combustion d'un alcool)}
\begin{enumerate}[label=\arabic*)]
\item $m_C = 3{,}52\times\dfrac{12}{44} = \SI{0.96}{\gram}$ ($n_C = \SI{0.08}{\mol}$) ;
$m_H = 1{,}80\times\dfrac{2}{18} = \SI{0.20}{\gram}$ ($n_H = \SI{0.20}{\mol}$) ;
$m_O = 1{,}48 - 0{,}96 - 0{,}20 = \SI{0.32}{\gram}$ ($n_O = \SI{0.02}{\mol}$).
$n_C : n_H : n_O = 4 : 10 : 1$ : formule $\ce{C4H10O}$ (un atome d'oxygène).
\item Isomères : butan-1-ol (I), butan-2-ol (II), 2-méthylpropan-1-ol (I),
2-méthylpropan-2-ol (III).
\item $A$ s'oxyde en un composé donnant le test de Tollens positif : c'est un
\emph{aldéhyde}, donc $A$ est un alcool \textbf{primaire}, le butan-1-ol. Oxydation en
excès : $\ce{CH3-CH2-CH2-CH2OH -> CH3-CH2-CH2-CHO -> CH3-CH2-CH2-COOH}$ (acide butanoïque).
\end{enumerate}

\corrige{du sujet 1}
\begin{enumerate}[label=\arabic*)]
\item $A$ réagit avec le sodium en libérant $\ce{H2}$ : c'est un alcool. Son oxydation
donne $B$ qui réduit la liqueur de Fehling, donc $B$ est un aldéhyde : $A$ est un
alcool \textbf{primaire}.
\item Alcools primaires en $\ce{C4H10O}$ : butan-1-ol $\ce{CH3-CH2-CH2-CH2OH}$ et
2-méthylpropan-1-ol $\ce{(CH3)2CH-CH2OH}$.
\item $\ce{2 R-CH2OH + 2 Na -> 2 R-CH2ONa + H2}$.
\item Oxydation (excès) : butan-1-ol $\to$ butanal $\to$ \textbf{acide butanoïque}
$\ce{CH3-CH2-CH2-COOH}$.
\end{enumerate}

\corrige{du sujet 2}
\begin{enumerate}[label=\arabic*)]
\item L'estérification $\ce{R-OH + CH3-COOH -> CH3-COO-R + H2O}$ conserve la quantité
de matière : $n(\text{alcool}) = n(\text{ester})$. De plus $M_{\text{ester}} =
M_{\text{alcool}} + 42$ (on remplace H par $\ce{CH3CO}$, soit $+60-18$).
\item $\dfrac{6{,}0}{M} = \dfrac{10{,}2}{M+42}$ donne $6{,}0\,(M+42) = 10{,}2\,M$,
d'où $M = \SI{60}{\gram\per\mol}$ : alcool primaire $\ce{C3H8O}$, le \textbf{propan-1-ol}.
\item $\ce{CH3-CH2-CH2OH + CH3-COOH -> CH3-COO-CH2-CH2-CH3 + H2O}$ (éthanoate de propyle).
\end{enumerate}

\section{Chapitre 6 — Les composés carbonylés}

\corrige{de l'exercice ★★★ (combustion d'un composé carbonylé)}
\begin{enumerate}[label=\arabic*)]
\item $m_C = 3{,}30\times\dfrac{12}{44} = \SI{0.90}{\gram}$ ($n_C = \SI{0.075}{\mol}$) ;
$m_H = 1{,}35\times\dfrac{2}{18} = \SI{0.15}{\gram}$ ($n_H = \SI{0.15}{\mol}$) ;
$m_O = 1{,}45 - 0{,}90 - 0{,}15 = \SI{0.40}{\gram}$ ($n_O = \SI{0.025}{\mol}$).
$n_C : n_H : n_O = 3 : 6 : 1$ : formule $\ce{C3H6O}$ ($M = \SI{58}{\gram\per\mol}$).
\item $A$ ne réduit pas la liqueur de Fehling : ce n'est pas un aldéhyde, c'est une
cétone : la \textbf{propanone} $\ce{CH3-CO-CH3}$.
\item $A$ provient de l'oxydation ménagée du propan-2-ol :
$\ce{CH3-CHOH-CH3 -> CH3-CO-CH3 + H2O}$ (par $\ce{[O]}$).
\end{enumerate}

\corrige{du sujet 1}
\begin{enumerate}[label=\arabic*)]
\item $B$ ne donne pas de précipité avec la 2,4-DNPH : il n'a pas de groupe carbonyle,
c'est l'alcool, le butan-1-ol. $A$ et $C$ sont des composés carbonylés ; $A$ réduit la
liqueur de Fehling (aldéhyde) $= $ butanal ; $C$ ne la réduit pas (cétone) $= $ butan-2-one.
\item $A$ : butanal $\ce{CH3-CH2-CH2-CHO}$ ; $B$ : butan-1-ol $\ce{CH3-CH2-CH2-CH2OH}$ ;
$C$ : butan-2-one $\ce{CH3-CO-CH2-CH3}$.
\item $\ce{CH3-CH2-CH2-CH2OH -> CH3-CH2-CH2-CHO + H2O}$ (oxydation ménagée, oxydant en défaut).
\item $C$ est une \textbf{cétone} : elle ne subit pas d'oxydation ménagée (pas
d'atome d'hydrogène sur le carbone du carbonyle). Aucune réaction.
\end{enumerate}

\corrige{du sujet 2}
\begin{enumerate}[label=\arabic*)]
\item Le groupe carbonyle $\ce{C=O}$ comporte une double liaison polarisée (le carbone
porte une charge partielle $\delta^+$) : elle peut subir une addition nucléophile.
\item $\ce{CH3-CO-CH3 + HCN -> (CH3)2C(OH)(CN)}$ (2-hydroxy-2-méthylpropanenitrile).
\end{enumerate}

\section{Chapitre 7 — Les acides carboxyliques et leurs dérivés}

\corrige{de l'estérification à l'équilibre}
\begin{enumerate}[label=\arabic*)]
\item $\ce{CH3-COOH + C2H5-OH <=> CH3-COO-C2H5 + H2O}$. Tableau (en mol) : à l'état
initial $1$ ; $1$ ; $0$ ; $0$ ; à l'équilibre $0{,}33$ ; $0{,}33$ ; $0{,}67$ ; $0{,}67$.
\item Rendement : $\rho = \dfrac{0{,}67}{1} = \SI{67}{\percent}$.
\item $K = \dfrac{[\text{ester}][\text{eau}]}{[\text{acide}][\text{alcool}]} =
\dfrac{0{,}67\times0{,}67}{0{,}33\times0{,}33} \approx 4{,}1$.
\end{enumerate}

\corrige{du sujet 1}
\begin{enumerate}[label=\arabic*)]
\item Acide $\ce{C_nH_{2n}O2}$ : $14n + 32 = 74$, $n = 3$ : acide propanoïque
$\ce{CH3-CH2-COOH}$.
\item $\ce{CH3-CH2-COOH + CH3-CH2-CH2OH <=> CH3-CH2-COO-CH2-CH2-CH3 + H2O}$
(propanoate de propyle).
\item Chlorure d'acyle : chlorure de propanoyle $\ce{CH3-CH2-COCl}$.
$\ce{CH3-CH2-COCl + CH3-CH2-CH2OH -> CH3-CH2-COO-CH2-CH2-CH3 + HCl}$.
\item $M(\text{ester}) = \SI{116}{\gram\per\mol}$, donc $n(\text{ester}) =
\dfrac{11{,}6}{116} = \SI{0.10}{\mol}$. La réaction étant totale,
$n(\text{chlorure}) = \SI{0.10}{\mol}$.
\end{enumerate}

\corrige{du sujet 2}
\begin{enumerate}[label=\arabic*)]
\item $\ce{AH + HO- -> A- + H2O}$.
\item À l'équivalence $n(\ce{AH}) = n(\ce{HO-}) = C_b V_{bE} = 0{,}10\times0{,}016 =
\SI{1.6e-3}{\mol}$. Concentration : $C_a = \dfrac{1{,}6\times10^{-3}}{0{,}020} = \SI{0.08}{\mol\per\litre}$.
\item $M = \dfrac{m}{n} = \dfrac{0{,}096}{1{,}6\times10^{-3}} = \SI{60}{\gram\per\mol}$ :
$A$ est l'\textbf{acide éthanoïque} $\ce{CH3-COOH}$ ($\ce{C2H4O2}$).
\end{enumerate}

\section{Chapitre 8 — Les amines}

\corrige{de l'exercice ★★★ (combustion d'une amine)}
\begin{enumerate}[label=\arabic*)]
\item $n(A) = \dfrac{4{,}5}{45} = \SI{0.1}{\mol}$.
$m_C = 8{,}8\times\dfrac{12}{44} = \SI{2.4}{\gram}$ soit $n_C = \SI{0.2}{\mol}$ (2 C par molécule) ;
$m_H = 6{,}3\times\dfrac{2}{18} = \SI{0.7}{\gram}$ soit $n_H = \SI{0.7}{\mol}$ (7 H par molécule) ;
$m_N = 4{,}5 - 2{,}4 - 0{,}7 = \SI{1.4}{\gram}$ soit $n_N = \SI{0.1}{\mol}$ (1 N par molécule).
Formule brute : $\ce{C2H7N}$.
\item Isomères : éthylamine $\ce{CH3-CH2-NH2}$ (primaire) et
$N$-méthylméthanamine $\ce{(CH3)2NH}$ (secondaire).
\item $\ce{CH3-CH2-NH2 + HCl -> CH3-CH2-NH3+ + Cl-}$ (chlorure d'éthylammonium).
\end{enumerate}

\corrige{du sujet}
\begin{enumerate}[label=\arabic*)]
\item Isomères de $\ce{C2H7N}$ : éthylamine $\ce{CH3-CH2-NH2}$ (primaire) et
$N$-méthylméthanamine $\ce{(CH3)2NH}$ (secondaire).
\item $\ce{CH3-CH2-NH2 + H2O <=> CH3-CH2-NH3+ + HO-}$ : il se forme l'ion
éthylammonium et l'ion hydroxyde (solution basique).
\item Dosage : $n(\text{amine}) = C V = 0{,}10\times0{,}020 = \SI{2.0e-3}{\mol}$.
$\ce{CH3-CH2-NH2 + H3O+ -> CH3-CH2-NH3+ + H2O}$. À l'équivalence
$n(\ce{HCl}) = n(\text{amine})$, d'où $V_{aE} = \dfrac{2{,}0\times10^{-3}}{0{,}10} = \SI{20}{\milli\litre}$.
\end{enumerate}

\section{Chapitre 9 — Les acides α-aminés et les protéines}

\corrige{de l'exercice ★★★ ($\ce{C3H7O2N}$)}
\begin{enumerate}[label=\arabic*)]
\item $M = 3\times12 + 7 + 2\times16 + 14 = \SI{89}{\gram\per\mol}$.
\item Acide $\alpha$-aminé en $\ce{C3}$ : $\ce{CH3-CH(NH2)-COOH}$, l'\textbf{alanine}
(acide 2-aminopropanoïque).
\item Le carbone central porte quatre groupes différents ($\ce{H}$, $\ce{NH2}$,
$\ce{CH3}$, $\ce{COOH}$) : c'est un carbone asymétrique, l'alanine est \textbf{chirale}
(deux énantiomères).
\item Zwitterion : $\ce{CH3-CH(NH3+)-COO-}$. En milieu acide :
$\ce{-COO- + H3O+ -> -COOH + H2O}$ ; en milieu basique :
$\ce{-NH3+ + HO- -> -NH2 + H2O}$.
\end{enumerate}

\corrige{du sujet}
\begin{enumerate}[label=\arabic*)]
\item En supposant un seul atome d'azote : $M = \dfrac{14}{0{,}157} \approx \SI{89}{\gram\per\mol}$.
\item $\ce{C_xH_yO2N}$ : $12x + y + 32 + 14 = 89$, soit $12x + y = 43$. Pour un composé
saturé, $x = 3$ et $y = 7$ : formule $\ce{C3H7O2N}$.
\item Formules possibles : l'alanine $\ce{CH3-CH(NH2)-COOH}$ (acide $\alpha$-aminé) et
l'acide 3-aminopropanoïque $\ce{H2N-CH2-CH2-COOH}$.
\item Seule l'\textbf{alanine} possède un carbone asymétrique : elle est chirale (active
sur la lumière polarisée). L'acide 3-aminopropanoïque ne l'est pas.
\end{enumerate}

\section{Chapitre 10 — La cinétique chimique}

\corrige{du sujet ($\ce{S2O8^2-}/\ce{I-}$)}
\begin{enumerate}[label=\arabic*)]
\item La courbe $[\ce{I2}] = f(t)$ croît puis atteint un palier (réaction terminée).
\item La vitesse de formation du diiode est $v = \dfrac{\mathrm{d}[\ce{I2}]}{\mathrm{d} t}$, pente de la
tangente à la courbe. À $t = 0$, la pente est maximale (de l'ordre de
$\SI{1.7}{\milli\mol\per\litre\per\minute}$) ; à $t = \SI{8}{\minute}$ elle est bien
plus faible (de l'ordre de $\SI{0.2}{\milli\mol\per\litre\per\minute}$).
\item La vitesse \textbf{diminue} au cours du temps : les réactifs se raréfient (le
facteur cinétique « concentration » diminue).
\item Concentration finale : $[\ce{I2}]_\infty = \SI{10}{\milli\mol\per\litre}$. Le temps
de demi-réaction correspond à $[\ce{I2}] = \SI{5}{\milli\mol\per\litre}$, atteint vers
$t_{1/2} \approx \SI{3}{\minute}$.
\item À température plus élevée, la réaction est plus rapide : la courbe monte plus
vite et atteint le palier plus tôt, pour la même valeur finale.
\end{enumerate}

\section{Chapitre 11 — Acides et bases}

\corrige{de l'exercice ★★★ (acide méthanoïque)}
\begin{enumerate}[label=\arabic*)]
\item $[\ce{H3O+}] = 10^{-2{,}4} = \SI{4.0e-3}{\mol\per\litre}$. Le coefficient de
dissociation $\alpha = \dfrac{4{,}0\times10^{-3}}{0{,}10} = 0{,}04 \ll 1$ : l'acide est faible.
\item $[\ce{H3O+}] = [\ce{HCOO-}] = \SI{4.0e-3}{\mol\per\litre}$ ;
$[\ce{HCOOH}] \approx 0{,}10 - 4{,}0\times10^{-3} \approx \SI{0.096}{\mol\per\litre}$ ;
$[\ce{HO-}] = \dfrac{K_e}{[\ce{H3O+}]} = \SI{2.5e-12}{\mol\per\litre}$.
\item $K_a = \dfrac{(4{,}0\times10^{-3})^2}{0{,}096} \approx \SI{1.7e-4}{}$, soit
$pK_a \approx 3{,}8$ (comparable à la valeur tabulée $3{,}75$).
\end{enumerate}

\corrige{du sujet}
\begin{enumerate}[label=\arabic*)]
\item $[\ce{H3O+}] = 10^{-3} = \SI{e-3}{\mol\per\litre}$ ;
$[\ce{HO-}] = \dfrac{K_e}{[\ce{H3O+}]} = \SI{e-11}{\mol\per\litre}$.
\item $\alpha = \dfrac{10^{-3}}{2{,}0\times10^{-2}} = 0{,}05 \ll 1$ : l'acide est faible.
\item $[\ce{H3O+}] = [\ce{A-}] = \SI{e-3}{\mol\per\litre}$ ;
$[\ce{AH}] = 2{,}0\times10^{-2} - 10^{-3} = \SI{1.9e-2}{\mol\per\litre}$.
\item $K_a = \dfrac{(10^{-3})^2}{1{,}9\times10^{-2}} \approx \SI{5.3e-5}{}$, soit $pK_a \approx 4{,}3$.
\item Pour un acide faible, une dilution au dixième fait augmenter le pH de
\emph{moins} d'une unité (l'équilibre se déplace dans le sens d'une dissociation accrue).
\end{enumerate}

\section{Chapitre 12 — Dosages et solutions tampons}

\corrige{du sujet (dosage de l'acide éthanoïque)}
\begin{enumerate}[label=\arabic*)]
\item Le saut de pH se situe entre $V_b = \SI{9.5}{\milli\litre}$ et
$\SI{10.5}{\milli\litre}$ : l'équivalence est à $V_{bE} = \SI{10}{\milli\litre}$,
pour $pH_E \approx 8{,}4$.
\item $C_a = \dfrac{C_b V_{bE}}{V_a} = \dfrac{0{,}10\times10}{10} = \SI{0.10}{\mol\per\litre}$.
\item À la demi-équivalence ($V_b = \SI{5}{\milli\litre}$), $pH = pK_a = 4{,}8$.
\item À l'équivalence, il ne reste que l'ion éthanoate $\ce{CH3COO-}$ (base faible) :
la solution est \textbf{basique}. On choisit la phénolphtaléine (zone $8{,}2$–$10$).
\item À la demi-équivalence, $[\ce{CH3COOH}] = [\ce{CH3COO-}]$ : c'est une
\textbf{solution tampon}, de pH égal au $pK_a$.
\end{enumerate}

\section{Chapitre 13 — Oxydoréduction : piles et électrolyse}

\corrige{du sujet (pile cuivre–argent)}
\begin{enumerate}[label=\arabic*)]
\item L'électrode de cuivre est le pôle négatif : elle est le siège de l'oxydation
(anode) $\ce{Cu -> Cu^2+ + 2 e-}$ ; à l'électrode d'argent (cathode, pôle $+$) :
$\ce{Ag+ + e- -> Ag}$. Bilan : $\ce{Cu + 2 Ag+ -> Cu^2+ + 2 Ag}$.
\item $Q = I\,t = 0{,}050\times(2\times3600) = \SI{360}{\coulomb}$.
\item $n(e^-) = \dfrac{Q}{F} = \dfrac{360}{96500} = \SI{3.73e-3}{\mol}$.
Cuivre consommé : $n(\ce{Cu}) = \dfrac{n(e^-)}{2}$, soit
$m(\ce{Cu}) = \dfrac{3{,}73\times10^{-3}}{2}\times63{,}5 \approx \SI{0.12}{\gram}$.
Argent déposé : $m(\ce{Ag}) = 3{,}73\times10^{-3}\times108 \approx \SI{0.40}{\gram}$.
\item L'électrode de cuivre \emph{perd} \SI{0.12}{\gram} ; l'électrode d'argent
\emph{gagne} \SI{0.40}{\gram}.
\end{enumerate}
